package ci.gouv.dgbf.system.payroll.server.api.shared;

import ci.gouv.dgbf.extension.server.service.api.request.FilterDto;

/**
 * {@inheritDoc}.
 *
 * @author Christian
 */
public interface FilterHasPayrollPeriodGroupIdentifier extends HasPayrollPeriodGroupIdentifier {

  /**
   * Cette méthode permet de mettre à jour identifiant groupe de période paie.
   *
   * @param filter {@link FilterDto}
   */
  default void updatePayrollPeriodGroupIdentifier(FilterDto filter) {
    setPayrollPeriodGroupIdentifier(PayrollPeriodGroupIdentifierFilter.get(filter));
  }

  /**
   * Cette méthode permet de mettre à jour identifiant groupe de période paie de
   * {@link FilterDto} .
   *
   * @param filter {@link FilterDto}
   */
  default void updateFilterPayrollPeriodGroupIdentifier(FilterDto filter) {
    PayrollPeriodGroupIdentifierFilter.set(filter, getPayrollPeriodGroupIdentifier());
  }

}
