package ci.gouv.dgbf.system.budgetaryclassification.server.api;

import java.util.stream.Stream;
import lombok.Getter;

/**
 * Cette classe représente les categorie d'acte.
 *
 * @author AKM
 *
 */
@Getter
public enum ActCategory {

  /**
   * {@link ActCategoryCode#DECREE_RESOURCES_INTEGRATION}.
   */
  DECREE_RESOURCES_INTEGRATION(ActCategoryCode.DECREE_RESOURCES_INTEGRATION,
      "Decret portant intégration de ressource", true),
  /**
   * {@link ActCategoryCode#INTERMINISTERIAL_ORDER_RESOURCES_INTEGRATION}.
   */
  INTERMINISTERIAL_ORDER_RESOURCES_INTEGRATION(
      ActCategoryCode.INTERMINISTERIAL_ORDER_RESOURCES_INTEGRATION,
      "Arrêté interministériel portant  intégration de ressource", true),

  /**
   * {@link ActCategoryCode#ORDER_RESOURCES_INTEGRATION}.
   */
  ORDER_RESOURCES_INTEGRATION(ActCategoryCode.ORDER_RESOURCES_INTEGRATION,
      "Arrêté portant  intégration de ressource", true),

  /**
   * {@link ActCategoryCode#ORDER_RESOURCES_UPDATE}.
   */
  ORDER_RESOURCES_UPDATE(ActCategoryCode.ORDER_RESOURCES_UPDATE,
      "Acte reglementaire modifiant la repartition des ressources", true),

  ;

  private ActCategory(String code, String name, boolean resource) {
    this.code = code;
    this.name = name;
    this.resource = resource;
  }

  private String code;
  private String name;
  private boolean resource;
  
  @Override
  public String toString() {
    return name;
  }

  /**
   * Cette méthode permet d'obtenir la categorie d'acte par code.
   *
   * @param code code
   * @return statut
   */
  public static ActCategory getByCode(String code) {
    return Stream.of(ActCategory.values()).filter(status -> status.getCode().equals(code))
        .findFirst().orElseThrow(IllegalArgumentException::new);
  }
}
