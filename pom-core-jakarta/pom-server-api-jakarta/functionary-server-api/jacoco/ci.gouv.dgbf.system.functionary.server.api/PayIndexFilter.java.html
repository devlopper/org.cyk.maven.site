package ci.gouv.dgbf.system.functionary.server.api;

import ci.gouv.dgbf.extension.server.service.api.AbstractIdentifiableFilter;
import ci.gouv.dgbf.extension.server.service.api.request.FilterDto;
import jakarta.json.bind.annotation.JsonbProperty;
import lombok.Getter;
import lombok.Setter;

/**
 * Cette classe représente le filtre de {@link PayIndexDto}.
 *
 * @author Christian
 *
 */
@Getter
@Setter
public class PayIndexFilter extends AbstractIdentifiableFilter {

  /**
   * Grade.
   */
  @JsonbProperty(value = JSON_GRADE_AS_STRING)
  String gradeAsString;

 

  /**
   * Cette méthode permet d'instancier un objet.
   */
  public PayIndexFilter() {
    super();
  }

  /**
   * Cette méthode permet d'instancier un objet.
   *
   * @param dto filtre
   */
  public PayIndexFilter(FilterDto dto) {
    super(dto);
  }

  @Override
  protected void doInitialize(FilterDto filter) {
    super.doInitialize(filter);
    gradeAsString = getGradeAsString(filter);
  
  }

  @Override
  protected void toDto(FilterDto filter) {
    super.toDto(filter);
    setGradeAsString(filter, gradeAsString);
  
  }

  /**
   * Cette méthode permet d'assigner {@link #gradeAsString}.
   *
   * @param filter filtre
   * @param gradeAsString {@link #gradeAsString}
   */
  public static void setGradeAsString(FilterDto filter, String gradeAsString) {
    set(filter, JSON_GRADE_AS_STRING, f -> f.getValueAsString(),
        f -> f.setValueAsString(gradeAsString));
  }

  /**
   * Cette méthode permet d'obtenir {@link #gradeAsString}.
   *
   * @param filter filtre
   * @return {@link #gradeAsString}
   */
  public static String getGradeAsString(FilterDto filter) {
    return get(filter, d -> d.getFieldValueAsStringByName(JSON_GRADE_AS_STRING));
  }

  
  /**
   * Identifiant json du {@link #gradeAsString}.
   */
  public static final String JSON_GRADE_AS_STRING = PayIndexDto.JSON_GRADE_AS_STRING;

  
}
