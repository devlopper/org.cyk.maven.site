package ci.gouv.dgbf.system.creditmanagementact.server.api;

import ci.gouv.dgbf.extension.server.service.api.entity.AbstractIdentifiableAuditableDto;
import jakarta.json.bind.annotation.JsonbProperty;
import lombok.EqualsAndHashCode;
import lombok.Getter;
import lombok.Setter;

/**
 * Cette classe représente la base de mouvement.
 *
 * @author Christian
 *
 */
@Getter
@Setter
@EqualsAndHashCode(callSuper = true)
public abstract class AbstractMovementDto extends AbstractIdentifiableAuditableDto {

  /**
   * Identifiant {@link ActDto}.
   */
  @JsonbProperty(JSON_ACT_IDENTIFIER)
  private String actIdentifier;
  
  /**
   * Représentation en chaine de caractère de {@link ActDto}.
   */
  @JsonbProperty(JSON_ACT_AS_STRING)
  private String actAsString;
  
  /**
   * montant.
   */
  @JsonbProperty(JSON_AMOUNT)
  private Long amount;
  
  /**
   * Représentation en chaine de caractères du montant.
   */
  @JsonbProperty(JSON_AMOUNT_AS_STRING)
  private String amountAsString;
  
  /**
   * Représentation en chaine de caractères du montant absolu.
   */
  @JsonbProperty(JSON_ABSOLUTE_AMOUNT_AS_STRING)
  private String absoluteAmountAsString;
  
  /**
   * Représentation en chaine de caractères du budget initial.
   */
  @JsonbProperty(JSON_INITIAL_BUDGET_AS_STRING)
  private String initialBudgetAsString;
  
  /**
   * Représentation en chaine de caractères du budget actuel.
   */
  @JsonbProperty(JSON_ACTUAL_BUDGET_AS_STRING)
  private String actualBudgetAsString;
  
  /**
   * Représentation en chaine de caractères du reste du budget exécuté.
   */
  @JsonbProperty(JSON_REMAINING_EXECUTED_BUDGET_AS_STRING)
  private String remainingExecutedBudgetAsString;
  
  /**
   * Est crédit.
   */
  @JsonbProperty(JSON_IS_CREDIT)
  private Boolean isCredit;
  
  /**
   * Identifiant json {@link AbstractMovementDto#actIdentifier}.
   */
  public static final String JSON_ACT_IDENTIFIER = "idActe";
  
  /**
   * Identifiant json {@link AbstractMovementDto#actAsString}.
   */
  public static final String JSON_ACT_AS_STRING = "acteChaine";
  
  /**
   * Identifiant json {@link AbstractMovementDto#amount}.
   */
  public static final String JSON_AMOUNT = "montant";
  
  /**
   * Identifiant json {@link AbstractMovementDto#amountAsString}.
   */
  public static final String JSON_AMOUNT_AS_STRING = "montantChaine";
  
  /**
   * Identifiant json {@link AbstractMovementDto#isCredit}.
   */
  public static final String JSON_IS_CREDIT = "estCredit";
  
  /**
   * Identifiant json {@link AbstractMovementDto#absoluteAmountAsString}.
   */
  public static final String JSON_ABSOLUTE_AMOUNT_AS_STRING = "montantAbsoluChaine";
  
  /**
   * Identifiant json {@link AbstractMovementDto#initialBudgetAsString}.
   */
  public static final String JSON_INITIAL_BUDGET_AS_STRING = "budgetInitialChaine";
  
  /**
   * Identifiant json {@link AbstractMovementDto#actualBudgetAsString}.
   */
  public static final String JSON_ACTUAL_BUDGET_AS_STRING = "budgetActuelChaine";
  
  /**
   * Identifiant json {@link AbstractMovementDto#remainingExecutedBudgetAsString}.
   */
  public static final String JSON_REMAINING_EXECUTED_BUDGET_AS_STRING = "resteBudgetExecute";
}
