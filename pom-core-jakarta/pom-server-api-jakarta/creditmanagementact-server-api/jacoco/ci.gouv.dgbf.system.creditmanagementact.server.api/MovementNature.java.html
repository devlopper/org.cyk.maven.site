package ci.gouv.dgbf.system.creditmanagementact.server.api;

import java.util.stream.Stream;
import lombok.Getter;

/**
 * Cette classe représente les natures de mouvement.
 *
 * @author Christian
 *
 */
@Getter
public enum MovementNature {

  /**
   * Virement.
   */
  PAYMENT(MovementNatureCode.PAYMENT, "Virement"),
  /**
   * Transfert.
   */
  TRANSFER(MovementNatureCode.TRANSFER, "Transfert"),

  ;

  private MovementNature(String code, String name) {
    this.code = code;
    this.name = name;
  }

  private String code;
  private String name;

  @Override
  public String toString() {
    return name;
  }

  /**
   * Cette méthode permet d'obtenir la nature de mouvement par code.
   *
   * @param code code
   * @return statut
   */
  public static MovementNature getByCode(String code) {
    return Stream.of(MovementNature.values())
        .filter(status -> status.getCode().equals(code)).findFirst()
        .orElseThrow(IllegalArgumentException::new);
  }
}
