package ci.gouv.dgbf.system.creditmanagementact.server.api;

import jakarta.json.bind.annotation.JsonbProperty;
import lombok.EqualsAndHashCode;
import lombok.Getter;
import lombok.Setter;

/**
 * Cette classe représente un mouvement de dépense.
 *
 * @author Christian
 *
 */
@Getter
@Setter
@EqualsAndHashCode(callSuper = true)
public class ExpenditureMovementDto extends AbstractMovementDto {

  /**
   * Identifiant du poste de dépense.
   */
  @JsonbProperty(JSON_ITEM_IDENTIFIER)
  private String itemIdentifier;

  /**
   * Représentation du poste de dépense en chaine de caractères.
   */
  @JsonbProperty(JSON_ITEM_AS_STRING)
  private String itemAsString;

  /**
   * Représentation de l'exercice du poste de dépense en chaine de caractères.
   */
  @JsonbProperty(JSON_ITEM_EXERCISE_AS_STRING)
  private String itemExerciseAsString;

  /**
   * Représentation de l'activité du poste de dépense en chaine de caractères.
   */
  @JsonbProperty(JSON_ITEM_ACTIVITY_AS_STRING)
  private String itemActivityAsString;

  /**
   * Code de l'activité.
   */
  @JsonbProperty(JSON_ITEM_ACTIVITY_CODE)
  private String itemActivityCode;

  /**
   * Représentation de la nature économique du poste de dépense en chaine de caractères.
   */
  @JsonbProperty(JSON_ITEM_ECONOMIC_NATURE_AS_STRING)
  private String itemEconomicNatureAsString;

  /**
   * Code de la nature économique.
   */
  @JsonbProperty(JSON_ITEM_ECONOMIC_NATURE_CODE)
  private String itemEconomicNatureCode;

  /**
   * Représentation de la source de financement du poste de dépense en chaine de caractères.
   */
  @JsonbProperty(JSON_ITEM_FUNDING_SOURCE_AS_STRING)
  private String itemFundingSourceAsString;

  /**
   * Représentation du bailleur du poste de dépense en chaine de caractères.
   */
  @JsonbProperty(JSON_ITEM_LESSOR_AS_STRING)
  private String itemLessorAsString;

  /**
   * Identifiant de la section.
   */
  @JsonbProperty(JSON_ITEM_SECTION_IDENTIFIER)
  private String itemSectionIdentifier;

  /**
   * Identifiant json {@link ExpenditureMovementDto#itemIdentifier}.
   */
  public static final String JSON_ITEM_IDENTIFIER = "idDepense";

  /**
   * Identifiant json {@link ExpenditureMovementDto#itemAsString}.
   */
  public static final String JSON_ITEM_AS_STRING = "depenseChaine";

  /**
   * Identifiant json {@link ExpenditureMovementDto#itemExerciseAsString}.
   */
  public static final String JSON_ITEM_EXERCISE_AS_STRING = "exerciceDepenseChaine";

  /**
   * Identifiant json {@link ExpenditureMovementDto#itemActivityAsString}.
   */
  public static final String JSON_ITEM_ACTIVITY_AS_STRING = "activiteDepenseChaine";

  /**
   * Identifiant json {@link ExpenditureMovementDto#itemActivityCode}.
   */
  public static final String JSON_ITEM_ACTIVITY_CODE = "activiteDepenseCode";

  /**
   * Identifiant json {@link ExpenditureMovementDto#itemEconomicNatureAsString}.
   */
  public static final String JSON_ITEM_ECONOMIC_NATURE_AS_STRING =
      "natureEconomiqueDepenseChaine";

  /**
   * Identifiant json {@link ExpenditureMovementDto#itemEconomicNatureCode}.
   */
  public static final String JSON_ITEM_ECONOMIC_NATURE_CODE =
      "natureEconomiqueDepenseCode";

  /**
   * Identifiant json {@link ExpenditureMovementDto#itemFundingSourceAsString}.
   */
  public static final String JSON_ITEM_FUNDING_SOURCE_AS_STRING =
      "sourceFinancementDepenseChaine";

  /**
   * Identifiant json {@link ExpenditureMovementDto#itemLessorAsString}.
   */
  public static final String JSON_ITEM_LESSOR_AS_STRING = "bailleurDepenseChaine";

  /**
   * Identifiant json {@link ExpenditureMovementDto#itemSectionIdentifier}.
   */
  public static final String JSON_ITEM_SECTION_IDENTIFIER = "idSection";

  /**
   * Nom pour lecture humaine de la classe.
   */
  public static final String NAME = "mouvement de dépense";

  /**
   * Nom au pluriel pour lecture humaine de la classe.
   */
  public static final String PLURAL_NAME = "mouvements de dépense";
}
