package ci.gouv.dgbf.system.creditmanagementact.server.api;

import jakarta.json.bind.annotation.JsonbProperty;
import lombok.EqualsAndHashCode;
import lombok.Getter;
import lombok.Setter;

/**
 * Cette classe représente un mouvement de dépense.
 *
 * @author Christian
 *
 */
@Getter
@Setter
@EqualsAndHashCode(callSuper = true)
public class ExpenditureMovementDto extends AbstractMovementDto {

  /**
   * Identifiant de la dépense.
   */
  @JsonbProperty(JSON_EXPENDITURE_IDENTIFIER)
  private String expenditureIdentifier;

  /**
   * Représentation de la dépense en chaine de caractères.
   */
  @JsonbProperty(JSON_EXPENDITURE_AS_STRING)
  private String expenditureAsString;

  /**
   * Représentation de l'exercice de la dépense en chaine de caractères.
   */
  @JsonbProperty(JSON_EXPENDITURE_EXERCISE_AS_STRING)
  private String expenditureExerciseAsString;

  /**
   * Représentation de l'activité de la dépense en chaine de caractères.
   */
  @JsonbProperty(JSON_EXPENDITURE_ACTIVITY_AS_STRING)
  private String expenditureActivityAsString;

  /**
   * Représentation de la nature économique de la dépense en chaine de caractères.
   */
  @JsonbProperty(JSON_EXPENDITURE_ECONOMIC_NATURE_AS_STRING)
  private String expenditureEconomicNatureAsString;

  /**
   * Représentation de la source de financement de la dépense en chaine de caractères.
   */
  @JsonbProperty(JSON_EXPENDITURE_FUNDING_SOURCE_AS_STRING)
  private String expenditureFundingSourceAsString;

  /**
   * Représentation du bailleur de la dépense en chaine de caractères.
   */
  @JsonbProperty(JSON_EXPENDITURE_LESSOR_AS_STRING)
  private String expenditureLessorAsString;

  /**
   * Identifiant json {@link ExpenditureMovementDto#expenditureIdentifier}.
   */
  public static final String JSON_EXPENDITURE_IDENTIFIER = "idDepense";

  /**
   * Identifiant json {@link ExpenditureMovementDto#expenditureAsString}.
   */
  public static final String JSON_EXPENDITURE_AS_STRING = "depenseChaine";

  /**
   * Identifiant json {@link ExpenditureMovementDto#expenditureExerciseAsString}.
   */
  public static final String JSON_EXPENDITURE_EXERCISE_AS_STRING = "exerciceDepenseChaine";

  /**
   * Identifiant json {@link ExpenditureMovementDto#expenditureActivityAsString}.
   */
  public static final String JSON_EXPENDITURE_ACTIVITY_AS_STRING = "activiteDepenseChaine";

  /**
   * Identifiant json {@link ExpenditureMovementDto#expenditureEconomicNatureAsString}.
   */
  public static final String JSON_EXPENDITURE_ECONOMIC_NATURE_AS_STRING =
      "natureEconomiqueDepenseChaine";

  /**
   * Identifiant json {@link ExpenditureMovementDto#expenditureFundingSourceAsString}.
   */
  public static final String JSON_EXPENDITURE_FUNDING_SOURCE_AS_STRING =
      "sourceFinancementDepenseChaine";

  /**
   * Identifiant json {@link ExpenditureMovementDto#expenditureLessorAsString}.
   */
  public static final String JSON_EXPENDITURE_LESSOR_AS_STRING = "bailleurDepenseChaine";

  /**
   * Nom pour lecture humaine de la classe.
   */
  public static final String NAME = "mouvement de dépense";

  /**
   * Nom au pluriel pour lecture humaine de la classe.
   */
  public static final String PLURAL_NAME = "mouvements de dépense";
}
