package ci.gouv.dgbf.system.creditmanagementact.server.api;

import ci.gouv.dgbf.extension.core.Core;
import java.util.Collections;
import java.util.List;
import java.util.Set;
import java.util.stream.Stream;
import lombok.Getter;

/**
 * Cette classe représente les statuts de {@link ActDto}.
 *
 * @author Christian
 *
 */
@Getter
public enum ActStatus {

  /**
   * Acte créé.
   */
  CREATED(ActStatusCode.CREATED, "Créé"),
  /**
   * Acte modifié.
   */
  UPDATED(ActStatusCode.UPDATED, "Modifié"),
  /**
   * Acte annulé.
   */
  CANCELLED(ActStatusCode.CANCELLED, "Annulé"),
  /**
   * Acte transmis.
   */
  TRANSMITTED(ActStatusCode.TRANSMITTED, "Transmis"),
  /**
   * Acte corrigé.
   */
  CORRECTED(ActStatusCode.CORRECTED, "Corrigé"),
  /**
   * Acte rejeté.
   */
  REJECTED(ActStatusCode.REJECTED, "Rejeté"),
  /**
   * Acte signé.
   */
  SIGNED(ActStatusCode.SIGNED, "Signé"),
  /**
   * Acte appliqué.
   */
  APPLIED(ActStatusCode.APPLIED, "Appliqué"),
  /**
   * Acte refusé.
   */
  REFUSED(ActStatusCode.REFUSED, "Refusé")

  ;

  private ActStatus(String code, String name) {
    this.code = code;
    this.name = name;
  }

  private String code;
  private String name;
  private Set<ActStatus> previous;

  @Override
  public String toString() {
    return name;
  }

  /**
   * Cette méthode permet d'obtenir le statut par code.
   *
   * @param code code
   * @return statut
   */
  public static ActStatus getByCode(String code) {
    return Stream.of(ActStatus.values()).filter(status -> status.getCode().equals(code)).findFirst()
        .orElseThrow(IllegalArgumentException::new);
  }

  static {
    CREATED.previous = Collections.emptySet();
    UPDATED.previous = Core.getSet(List.of(CREATED, UPDATED));
    CANCELLED.previous = Core.getSet(List.of(CREATED, UPDATED));
    TRANSMITTED.previous = Core.getSet(List.of(CREATED, UPDATED));
  }
}
