package ci.gouv.dgbf.system.creditmanagementact.server.api;

import ci.gouv.dgbf.extension.server.service.api.AbstractIdentifiableFilter;
import ci.gouv.dgbf.extension.server.service.api.request.FilterDto;
import lombok.Getter;
import lombok.Setter;

/**
 * Cette classe représente le filtre de {@link ExpenditureMovementDto}.
 *
 * @author Christian
 *
 */
@Getter
@Setter
public class ExpenditureMovementFilter extends AbstractIdentifiableFilter {
  /**
   * Identifiant de {@link ActDto}.
   */
  String actIdentifier;

  /**
   * Cette méthode permet d'instantier un objet.
   *
   * @param dto filtre
   */
  public ExpenditureMovementFilter(FilterDto dto) {
    super(dto);
  }

  /**
   * Cette méthode permet d'instantier un objet.
   */
  public ExpenditureMovementFilter() {}

  @Override
  protected void doInitialize(FilterDto filter) {
    super.doInitialize(filter);
    actIdentifier = getActIdentifier(filter);
  }

  @Override
  protected void toDto(FilterDto filter) {
    super.toDto(filter);
    setActIdentifier(filter, actIdentifier);
  }

  /**
   * Cette méthode permet d'assigner l'identifiant de {@link ActDto}.
   *
   * @param filter filtre
   * @param identifier identifiant
   */
  public static void setActIdentifier(FilterDto filter, String identifier) {
    set(filter, JSON_ACT_IDENTIFIER, f -> f.getValueAsString(),
        f -> f.setValueAsString(identifier));
  }

  /**
   * Cette méthode permet d'obtenir l'identifiant de {@link ActDto}.
   *
   * @param filter filtre
   * @return identifiant de {@link ActDto}
   */
  public static String getActIdentifier(FilterDto filter) {
    return get(filter, d -> d.getFieldValueAsStringByName(JSON_ACT_IDENTIFIER));
  }

  /**
   * Identifiant json champ {@link ExpenditureMovementFilter#actIdentifier}.
   */
  public static final String JSON_ACT_IDENTIFIER = ExpenditureMovementDto.JSON_ACT_IDENTIFIER;
}
