package ci.gouv.dgbf.system.creditmanagementact.server.api;

import jakarta.json.bind.annotation.JsonbProperty;
import lombok.EqualsAndHashCode;
import lombok.Getter;
import lombok.Setter;

/**
 * Cette classe représente un mouvement de ressource.
 *
 * @author Christian
 *
 */
@Getter
@Setter
@EqualsAndHashCode(callSuper = true)
public class ResourceMovementDto extends AbstractMovementDto {
  
  /**
   * Identifiant de la ressource.
   */
  @JsonbProperty(JSON_RESOURCE_IDENTIFIER)
  private String resourceIdentifier;

  /**
   * Représentation du poste de dépense en chaine de caractères.
   */
  @JsonbProperty(JSON_RESOURCE_AS_STRING)
  private String resourceAsString;

  /**
   * Représentation de l'exercice du poste de dépense en chaine de caractères.
   */
  @JsonbProperty(JSON_RESOURCE_EXERCISE_AS_STRING)
  private String resourceExerciseAsString;

  /**
   * Représentation de l'activité du poste de dépense en chaine de caractères.
   */
  @JsonbProperty(JSON_RESOURCE_ACTIVITY_AS_STRING)
  private String resourceActivityAsString;

  /**
   * Code de l'activité.
   */
  @JsonbProperty(JSON_RESOURCE_ACTIVITY_CODE)
  private String resourceActivityCode;

  /**
   * Représentation de la nature économique du poste de dépense en chaine de caractères.
   */
  @JsonbProperty(JSON_RESOURCE_ECONOMIC_NATURE_AS_STRING)
  private String resourceEconomicNatureAsString;

  /**
   * Code de la nature économique.
   */
  @JsonbProperty(JSON_RESOURCE_ECONOMIC_NATURE_CODE)
  private String resourceEconomicNatureCode;
  
  /**
   * Identifiant json {@link ResourceMovementDto#resourceIdentifier}.
   */
  public static final String JSON_RESOURCE_IDENTIFIER = "idRessource";

  /**
   * Identifiant json {@link ResourceMovementDto#resourceAsString}.
   */
  public static final String JSON_RESOURCE_AS_STRING = "resourceChaine";

  /**
   * Identifiant json {@link ResourceMovementDto#resourceExerciseAsString}.
   */
  public static final String JSON_RESOURCE_EXERCISE_AS_STRING = "exerciceResourceChaine";

  /**
   * Identifiant json {@link ResourceMovementDto#resourceActivityAsString}.
   */
  public static final String JSON_RESOURCE_ACTIVITY_AS_STRING = "activiteResourceChaine";

  /**
   * Identifiant json {@link ResourceMovementDto#resourceActivityCode}.
   */
  public static final String JSON_RESOURCE_ACTIVITY_CODE = "activiteResourceCode";

  /**
   * Identifiant json {@link ResourceMovementDto#resourceEconomicNatureAsString}.
   */
  public static final String JSON_RESOURCE_ECONOMIC_NATURE_AS_STRING =
      "natureEconomiqueResourceChaine";

  /**
   * Identifiant json {@link ResourceMovementDto#resourceEconomicNatureCode}.
   */
  public static final String JSON_RESOURCE_ECONOMIC_NATURE_CODE =
      "natureEconomiqueResourceCode";
  
  /**
   * Nom pour lecture humaine de la classe.
   */
  public static final String NAME = "mouvement de ressource";
  
  /**
   * Nom au pluriel pour lecture humaine de la classe.
   */
  public static final String PLURAL_NAME = "mouvements de ressource";
}
