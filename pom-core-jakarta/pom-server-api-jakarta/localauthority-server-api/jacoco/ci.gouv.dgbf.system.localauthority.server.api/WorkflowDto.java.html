package ci.gouv.dgbf.system.localauthority.server.api;

import ci.gouv.dgbf.extension.server.service.api.entity.AbstractIdentifiableCodableNamableAuditableDto;
import jakarta.json.bind.annotation.JsonbProperty;
import lombok.Getter;
import lombok.Setter;

/**
 * Cette classe représente un workflow.
 *
 * @author Christian
 *
 */
@Getter
@Setter
public class WorkflowDto extends AbstractIdentifiableCodableNamableAuditableDto {

  /**
   * Identifiant de {@link BudgetDto}.
   */
  @JsonbProperty(JSON_BUDGET_IDENTIFIER)
  private String budgetIdentifier;

  /**
   * Représentation en chaine de caractères de {@link BudgetDto}.
   */
  @JsonbProperty(JSON_BUDGET_AS_STRING)
  private String budgetAsString;

  /**
   * Ordre.
   */
  @JsonbProperty(JSON_ORDRE)
  private int ordre;

  /**
   * Représentation en chaine de caractères d'ordre.
   */
  @JsonbProperty(JSON_ORDRE_AS_STRING)
  private String ordreAsString;

  /**
   * Identitiant json {@link WorkflowDto#budgetIdentifier}.
   */
  public static final String JSON_BUDGET_IDENTIFIER = "idBudget";

  /**
   * Identitiant json {@link WorkflowDto#budgetAsString}.
   */
  public static final String JSON_BUDGET_AS_STRING = "budgetChaine";

  /**
   * Identifiant json {@link WorkflowDto#ordre}.
   */
  public static final String JSON_ORDRE = "ordre";


  /**
   * Identifiant json {@link WorkflowDto#ordreAsString}.
   */
  public static final String JSON_ORDRE_AS_STRING = "ordreChaine";

  /**
   * Nom pour lecture humaine de la classe.
   */
  public static final String NAME = "workflow";

  /**
   * Nom au pluriel pour lecture humaine de la classe.
   */
  public static final String PLURAL_NAME = "workflows";
}
