package ci.gouv.dgbf.system.localauthority.server.api;


import ci.gouv.dgbf.extension.core.Core;
import java.util.Collections;
import java.util.List;
import java.util.Set;
import java.util.stream.Stream;
import lombok.Getter;

/**
 * Cette classe représente les statuts de {@link TriennialProgramDto}.
 *
 * @author Christian
 *
 */
@Getter
public enum TriennialProgramStatus {

  /**
   * TriennalPrograme créé.
   */
  CREATED(TriennialProgramStatusCode.CREATED, "Créé", "Création"),


  /**
   * TriennalPrograme transmis.
   */
  TRANSMITTED(TriennialProgramStatusCode.TRANSMITTED, "Transmis", "Transmission"),
  /**
   * TriennalPrograme Accepté.
   */
  ACCEPTED(TriennialProgramStatusCode.ACCEPTED, "Accepté", "Validation"),
  /**
   * TriennalPrograme Retourné.
   */
  RETURNED(TriennialProgramStatusCode.RETURNED, "Retourné", "Rejet"),
  /**
   * TriennalPrograme corrigé.
   */
  CORRECTED(TriennialProgramStatusCode.CORRECTED, "Corrigé", "Correction"),
  /**
   * TriennalPrograme corrigé.
   */
  APPROVED(TriennialProgramStatusCode.APPROVED, "Approuvé", "Approbation"),;

  private TriennialProgramStatus(String code, String name, String action) {
    this.code = code;
    this.name = name;
    this.action = action;
  }

  private String code;
  private String name;
  private String action;
  private Set<TriennialProgramStatus> previous;

  @Override
  public String toString() {
    return name;
  }

  /**
   * Cette méthode permet d'obtenir le statut par code.
   *
   * @param code code
   * @return statut
   */
  public static TriennialProgramStatus getByCode(String code) {
    return Stream.of(TriennialProgramStatus.values())
        .filter(status -> status.getCode().equals(code)).findFirst()
        .orElseThrow(IllegalArgumentException::new);
  }

  static {
    CREATED.previous = Collections.emptySet();
    TRANSMITTED.previous = Core.getSet(List.of(CREATED));
    ACCEPTED.previous = Core.getSet(List.of(TRANSMITTED));
    RETURNED.previous = Core.getSet(List.of(ACCEPTED));
    CORRECTED.previous = Core.getSet(List.of(ACCEPTED));
    APPROVED.previous = Core.getSet(List.of(ACCEPTED));
  }
}
