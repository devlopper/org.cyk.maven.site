package ci.gouv.dgbf.system.gestbull.server.api;

import ci.gouv.dgbf.extension.server.service.api.entity.AbstractIdentifiableDto;
import jakarta.json.bind.annotation.JsonbProperty;
import lombok.EqualsAndHashCode;
import lombok.Getter;
import lombok.Setter;

/**
 * Cette classe représente un bulletin de salaire.
 *
 * @author Christian
 *
 */
@Getter
@Setter
@EqualsAndHashCode(callSuper = true)
public class PayslipDto extends AbstractIdentifiableDto {

  /**
   * Année.
   */
  @JsonbProperty(value = JSON_YEAR)
  private Integer year;

  /**
   * Mois.
   */
  @JsonbProperty(value = JSON_MONTH)
  private Integer month;

  /**
   * Matricule.
   */
  @JsonbProperty(value = JSON_REGISTRATION_NUMBER)
  private String registrationNumber;
  
  /**
   * Octects de fichier.
   */
  @JsonbProperty(value = JSON_FILE_BYTES)
  private byte[] fileBytes;
  
  /**
   * Identifiant JSON {@link #year}.
   */
  public static final String JSON_YEAR = "annee";
  
  /**
   * Identifiant JSON {@link #month}.
   */
  public static final String JSON_MONTH = "mois";
  
  /**
   * Identifiant JSON {@link #registrationNumber}.
   */
  public static final String JSON_REGISTRATION_NUMBER = "matricule";
  
  /**
   * Identifiant JSON {@link #fileBytes}.
   */
  public static final String JSON_FILE_BYTES = "octetsFichier";
}
