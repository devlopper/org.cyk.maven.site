package ci.gouv.dgbf.system.gestbull.server.api;

import ci.gouv.dgbf.extension.server.service.api.AbstractIdentifiableFilter;
import ci.gouv.dgbf.extension.server.service.api.request.FilterDto;
import jakarta.json.bind.annotation.JsonbProperty;
import lombok.Getter;
import lombok.Setter;

/**
 * Cette classe représente le filtre de {@link PayslipDto}.
 *
 * @author Christian
 *
 */
@Getter
@Setter
public class PayslipFilter extends AbstractIdentifiableFilter {

  /**
   * Année.
   */
  @JsonbProperty(value = JSON_YEAR)
  private Integer year;

  /**
   * Mois.
   */
  @JsonbProperty(value = JSON_MONTH)
  private Integer month;

  /**
   * Matricule.
   */
  @JsonbProperty(value = JSON_REGISTRATION_NUMBER)
  private String registrationNumber;

  @Override
  protected void doInitialize(FilterDto filter) {
    super.doInitialize(filter);
    year = getYear(filter);
    month = getMonth(filter);
    registrationNumber = getRegistrationNumber(filter);
  }

  @Override
  protected void toDto(FilterDto filter) {
    super.toDto(filter);
    setYear(filter, year);
    setMonth(filter, month);
    setRegistrationNumber(filter, registrationNumber);
  }

  /**
   * Cette méthode permet d'obtenir l'année.
   *
   * @param filter filtre
   * @return année
   */
  public static Integer getYear(FilterDto filter) {
    return get(filter, d -> d.getFieldValueAsIntegerByName(JSON_YEAR));
  }

  /**
   * Cette méthode permet d'assigner l'année.
   *
   * @param filter filtre
   * @param year année
   */
  public static void setYear(FilterDto filter, Integer year) {
    set(filter, JSON_YEAR, f -> f.getValueAsInteger(), f -> f.setValueAsInteger(year));
  }

  /**
   * Cette méthode permet d'obtenir le mois.
   *
   * @param filter filtre
   * @return mois
   */
  public static Integer getMonth(FilterDto filter) {
    return get(filter, d -> d.getFieldValueAsIntegerByName(JSON_MONTH));
  }

  /**
   * Cette méthode permet d'assigner le mois.
   *
   * @param filter filtre
   * @param month mois
   */
  public static void setMonth(FilterDto filter, Integer month) {
    set(filter, JSON_MONTH, f -> f.getValueAsInteger(), f -> f.setValueAsInteger(month));
  }

  /**
   * Cette méthode permet d'obtenir le matricule.
   *
   * @param filter filtre
   * @return matricule
   */
  public static String getRegistrationNumber(FilterDto filter) {
    return get(filter, d -> d.getFieldValueAsStringByName(JSON_REGISTRATION_NUMBER));
  }

  /**
   * Cette méthode permet d'assigner le matricule.
   *
   * @param filter filtre
   * @param registrationNumber matricule
   */
  public static void setRegistrationNumber(FilterDto filter, String registrationNumber) {
    set(filter, JSON_REGISTRATION_NUMBER, f -> f.getValueAsString(),
        f -> f.setValueAsString(registrationNumber));
  }

  /**
   * {@link PayslipDto#JSON_YEAR}.
   */
  public static final String JSON_YEAR = PayslipDto.JSON_YEAR;

  /**
   * {@link PayslipDto#JSON_MONTH}.
   */
  public static final String JSON_MONTH = PayslipDto.JSON_MONTH;

  /**
   * {@link PayslipDto#JSON_REGISTRATION_NUMBER}.
   */
  public static final String JSON_REGISTRATION_NUMBER = PayslipDto.JSON_REGISTRATION_NUMBER;
}
