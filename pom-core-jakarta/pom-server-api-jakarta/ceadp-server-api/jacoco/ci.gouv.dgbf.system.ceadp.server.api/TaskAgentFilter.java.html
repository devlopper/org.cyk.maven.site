package ci.gouv.dgbf.system.ceadp.server.api;

import ci.gouv.dgbf.extension.server.service.api.AbstractIdentifiableFilter;
import ci.gouv.dgbf.extension.server.service.api.request.FilterDto;
import lombok.Getter;
import lombok.Setter;

/**
 * Cette classe représente le filtre de {@link TaskAgentDto}.
 *
 * @author Christian
 *
 */
@Getter
@Setter
public class TaskAgentFilter extends AbstractIdentifiableFilter {
  /**
   * Identifiant de {@link TaskDto}.
   */
  String taskIdentifier;

  /**
   * Cette méthode permet d'instantier un objet.
   *
   * @param dto filtre
   */
  public TaskAgentFilter(FilterDto dto) {
    super(dto);
  }

  /**
   * Cette méthode permet d'instantier un objet.
   */
  public TaskAgentFilter() {}

  @Override
  protected void doInitialize(FilterDto filter) {
    super.doInitialize(filter);
    taskIdentifier = getTaskIdentifier(filter);
  }

  @Override
  protected void toDto(FilterDto filter) {
    super.toDto(filter);
    setTaskIdentifier(filter, taskIdentifier);
  }

  /**
   * Cette méthode permet d'assigner l'identifiant de {@link TaskDto}.
   *
   * @param filter filtre
   * @param identifier identifiant
   */
  public static void setTaskIdentifier(FilterDto filter, String identifier) {
    set(filter, JSON_TASK_IDENTIFIER, f -> f.getValueAsString(),
        f -> f.setValueAsString(identifier));
  }

  /**
   * Cette méthode permet d'obtenir l'identifiant de {@link TaskDto}.
   *
   * @param filter filtre
   * @return identifiant de {@link TaskDto}
   */
  public static String getTaskIdentifier(FilterDto filter) {
    return get(filter, d -> d.getFieldValueAsStringByName(JSON_TASK_IDENTIFIER));
  }

  /**
   * Identifiant json champ {@link TaskAgentFilter#taskIdentifier}.
   */
  public static final String JSON_TASK_IDENTIFIER = TaskAgentDto.JSON_TASK_IDENTIFIER;
}
