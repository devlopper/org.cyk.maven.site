package ci.gouv.dgbf.system.ceadp.server.api;

import java.util.Collections;
import java.util.Set;
import java.util.stream.Stream;
import lombok.Getter;

/**
 * Cette classe représente les statuts de {@link FolderDto}.
 *
 * @author Christian
 *
 */
@Getter
public enum FolderStatus {

  /**
   * Cette énumération permet de signifier que le dossier est créé.
   */
  CREATED(FolderStatusCode.CREATED, "Créé"),

  /**
   * Cette énumération permet de signifier que le dossier est validé.
   */
  VALIDATED(FolderStatusCode.VALIDATED, "Validé")

  ;

  private FolderStatus(String code, String name) {
    this.code = code;
    this.name = name;
  }

  private String code;
  private String name;
  private Set<FolderStatus> previous;

  @Override
  public String toString() {
    return name;
  }

  /**
   * Cette méthode permet d'obtenir un statut par code.
   *
   * @param code ode
   * @return statut
   */
  public static FolderStatus getByCode(String code) {
    return Stream.of(FolderStatus.values()).filter(status -> status.getCode().equals(code))
        .findFirst().orElseThrow(IllegalArgumentException::new);
  }

  static {
    CREATED.previous = Collections.emptySet();
    VALIDATED.previous = Set.of(CREATED);
  }
}
