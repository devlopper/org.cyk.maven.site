package ci.gouv.dgbf.system.payrollbudget.server.api;

import ci.gouv.dgbf.extension.server.service.api.AbstractIdentifiableFilter;
import ci.gouv.dgbf.extension.server.service.api.request.FilterDto;
import lombok.Getter;
import lombok.Setter;

/**
 * Cette classe représente la base de filtre par {@link ExerciseDto}.
 *
 * @author Christian
 *
 */
@Getter
@Setter
public abstract class AbstractExerciseFilter extends AbstractIdentifiableFilter {

  /**
   * Identifiant {@link BudgetDto}.
   */
  String budgetIdentifier;

  /**
   * Identifiant {@link ExerciseDto}.
   */
  @Deprecated(forRemoval = true)
  String exerciseIdentifier;

  /**
   * Cette méthode permet d'instantier un objet.
   *
   * @param dto filtre
   */
  public AbstractExerciseFilter(FilterDto dto) {
    super(dto);
  }

  /**
   * Cette méthode permet d'instantier un objet.
   */
  public AbstractExerciseFilter() {}

  @Override
  protected void doInitialize(FilterDto filter) {
    super.doInitialize(filter);
    exerciseIdentifier = getExerciseIdentifier(filter);
    budgetIdentifier = getBudgetIdentifier(filter);
  }

  @Override
  protected void toDto(FilterDto filter) {
    super.toDto(filter);
    setBudgetIdentifier(filter, budgetIdentifier);
    setExerciseIdentifier(filter, exerciseIdentifier);
  }

  /**
   * Cette méthode permet d'obtenir l'identifiant de {@link BudgetDto}.
   *
   * @param filter filtre
   * @return identifiant de {@link BudgetDto}
   */
  public static String getBudgetIdentifier(FilterDto filter) {
    return get(filter, d -> d.getFieldValueAsStringByName(JSON_BUDGET_IDENTIFIER));
  }

  /**
   * Cette méthode permet d'assigner l'identifiant de {@link BudgetDto}.
   *
   * @param filter filtre
   * @param identifier identifiant de l'identifiant de {@link BudgetDto}
   */
  public static void setBudgetIdentifier(FilterDto filter, String identifier) {
    set(filter, JSON_BUDGET_IDENTIFIER, f -> f.getValueAsString(),
        f -> f.setValueAsString(identifier));
  }

  /**
   * Cette méthode permet d'obtenir l'identifiant de {@link ExerciseDto}.
   *
   * @param filter filtre
   * @return identifiant de {@link ExerciseDto}
   */
  @Deprecated(forRemoval = true)
  public static String getExerciseIdentifier(FilterDto filter) {
    return get(filter, d -> d.getFieldValueAsStringByName(JSON_EXERCISE_IDENTIFIER));
  }

  /**
   * Cette méthode permet d'assigner l'identifiant de {@link ExerciseDto}.
   *
   * @param filter filtre
   * @param identifier identifiant de l'identifiant de {@link ExerciseDto}
   */
  @Deprecated(forRemoval = true)
  public static void setExerciseIdentifier(FilterDto filter, String identifier) {
    set(filter, JSON_EXERCISE_IDENTIFIER, f -> f.getValueAsString(),
        f -> f.setValueAsString(identifier));
  }

  /**
   * Identifiant json {@link #budgetIdentifier}.
   */
  public static final String JSON_BUDGET_IDENTIFIER = BudgetDto.JSON_THIS_IDENTIFIER;

  /**
   * Identifiant json {@link #exerciseIdentifier}.
   */
  public static final String JSON_EXERCISE_IDENTIFIER = ExerciseDto.JSON_THIS_IDENTIFIER;
}
