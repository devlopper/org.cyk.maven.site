package ci.gouv.dgbf.system.payrollbudget.server.api;

import ci.gouv.dgbf.extension.server.service.api.request.FilterDto;
import lombok.Getter;
import lombok.Setter;

/**
 * Cette classe représente le filtre de {@link AdministrativeUnitEmployeeCategoryDto}.
 *
 * @author Christian
 *
 */
@Getter
@Setter
public class AdministrativeUnitEmployeeCategoryFilter
    extends AbstractExerciseAdministrativeUnitFilter {

  /**
   * Identifiant {@link EmployeeCategoryDto}.
   */
  String employeeCategoryIdentifier;

  /**
   * Cette méthode permet d'instantier un objet.
   *
   * @param dto filtre
   */
  public AdministrativeUnitEmployeeCategoryFilter(FilterDto dto) {
    super(dto);
  }

  /**
   * Cette méthode permet d'instantier un objet.
   */
  public AdministrativeUnitEmployeeCategoryFilter() {}

  @Override
  protected void doInitialize(FilterDto filter) {
    super.doInitialize(filter);
    employeeCategoryIdentifier = getEmployeeCategoryIdentifier(filter);
  }

  @Override
  protected void toDto(FilterDto filter) {
    super.toDto(filter);
    setEmployeeCategoryIdentifier(filter, employeeCategoryIdentifier);
  }

  /**
   * Cette méthode permet d'obtenir l'identifiant de {@link AdministrativeUnitDto}.
   *
   * @param filter filtre
   * @return identifiant {@link AdministrativeUnitDto}
   */
  public static String getEmployeeCategoryIdentifier(FilterDto filter) {
    return get(filter, d -> d.getFieldValueAsStringByName(JSON_EMPLOYEE_CATEGORY_IDENTIFIER));
  }

  /**
   * Cette méthode permet d'assigner l'identifiant de l'unité administrative.
   *
   * @param filter filtre
   * @param identifier identifiant {@link AdministrativeUnitDto}
   */
  public static void setEmployeeCategoryIdentifier(FilterDto filter, String identifier) {
    set(filter, JSON_EMPLOYEE_CATEGORY_IDENTIFIER, f -> f.getValueAsString(),
        f -> f.setValueAsString(identifier));
  }

  /**
   * Identifiant json {@link #employeeCategoryIdentifier}.
   */
  public static final String JSON_EMPLOYEE_CATEGORY_IDENTIFIER =
      EmployeeCategoryDto.JSON_THIS_IDENTIFIER;
}
