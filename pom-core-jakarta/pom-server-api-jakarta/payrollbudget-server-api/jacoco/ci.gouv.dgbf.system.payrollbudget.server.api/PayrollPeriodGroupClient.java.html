/**

    GetByIdentifier<PayrollPeriodGroupDto>, GetMany<PayrollPeriodGroupGetManyResponseDto> {

  @Override

  /**

    request.setProjection(projection);
    request.setFilter(filter);
    request.setAuditWho(auditWho);
