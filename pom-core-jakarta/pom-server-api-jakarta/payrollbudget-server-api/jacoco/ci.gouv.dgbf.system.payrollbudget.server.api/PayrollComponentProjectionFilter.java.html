package ci.gouv.dgbf.system.payrollbudget.server.api;

import ci.gouv.dgbf.extension.server.service.api.request.FilterDto;
import lombok.Getter;
import lombok.Setter;

/**
 * Cette classe représente le filtre de {@link PayrollComponentProjectionDto}.
 *
 * @author Christian
 */
@Getter
@Setter
public class PayrollComponentProjectionFilter
    extends AbstractExerciseAdministrativeUnitEmployeeFilter {

  /**
   * Identifiant de {@link PayrollComponentDto}.
   */
  String payrollComponentIdentifier;

  /**
   * Identifiant de {@link PayrollPeriodGroupDto}.
   */
  String payrollPeriodGroupIdentifier;

  /**
   * Cette méthode permet d'instantier un objet.
   *
   * @param dto filtre
   */
  public PayrollComponentProjectionFilter(FilterDto dto) {
    super(dto);
  }

  /**
   * Cette méthode permet d'instantier un objet.
   */
  public PayrollComponentProjectionFilter() {}

  @Override
  protected void doInitialize(FilterDto filter) {
    super.doInitialize(filter);
    payrollComponentIdentifier = getPayrollComponentIdentifier(filter);
    payrollPeriodGroupIdentifier = getPayrollPeriodGroupIdentifier(filter);
  }

  @Override
  protected void toDto(FilterDto filter) {
    super.toDto(filter);
    setPayrollComponentIdentifier(filter, payrollComponentIdentifier);
    setPayrollPeriodGroupIdentifier(filter, payrollPeriodGroupIdentifier);
  }

  /**
   * Cette méthode permet d'obtenir l'identifiant de {@link PayrollComponentDto}.
   *
   * @param filter filtre
   * @return identifiant de {@link PayrollComponentDto}
   */
  public static String getPayrollComponentIdentifier(FilterDto filter) {
    return get(filter, d -> d.getFieldValueAsStringByName(JSON_PAYROLL_COMPONENT_IDENTIFIER));
  }

  /**
   * Cette méthode permet d'assigner l'identifiant de {@link PayrollComponentDto}.
   *
   * @param filter filtre
   * @param identifier identifiant de l'identifiant de {@link PayrollComponentDto}
   */
  public static void setPayrollComponentIdentifier(FilterDto filter, String identifier) {
    set(filter, JSON_PAYROLL_COMPONENT_IDENTIFIER, f -> f.getValueAsString(),
        f -> f.setValueAsString(identifier));
  }

  /**
   * Cette méthode permet d'obtenir l'identifiant de {@link PayrollPeriodGroupDto}.
   *
   * @param filter filtre
   * @return identifiant de {@link PayrollPeriodGroupDto}
   */
  public static String getPayrollPeriodGroupIdentifier(FilterDto filter) {
    return get(filter, d -> d.getFieldValueAsStringByName(JSON_PAYROLL_PERIOD_GROUP_IDENTIFIER));
  }

  /**
   * Cette méthode permet d'assigner l'identifiant de {@link PayrollPeriodGroupDto}.
   *
   * @param filter filtre
   * @param identifier identifiant de l'identifiant de {@link PayrollPeriodGroupDto}
   */
  public static void setPayrollPeriodGroupIdentifier(FilterDto filter, String identifier) {
    set(filter, JSON_PAYROLL_PERIOD_GROUP_IDENTIFIER, f -> f.getValueAsString(),
        f -> f.setValueAsString(identifier));
  }

  /**
   * Identifiant json {@link #payrollComponentIdentifier}.
   */
  public static final String JSON_PAYROLL_COMPONENT_IDENTIFIER =
      PayrollComponentDto.JSON_THIS_IDENTIFIER;

  /**
   * Identifiant json {@link #payrollPeriodGroupIdentifier}.
   */
  public static final String JSON_PAYROLL_PERIOD_GROUP_IDENTIFIER =
      PayrollPeriodGroupDto.JSON_THIS_IDENTIFIER;
}
