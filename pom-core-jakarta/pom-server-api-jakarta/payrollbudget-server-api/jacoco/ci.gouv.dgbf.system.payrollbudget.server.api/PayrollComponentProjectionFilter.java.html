package ci.gouv.dgbf.system.payrollbudget.server.api;


import ci.gouv.dgbf.extension.server.service.api.AbstractIdentifiableFilter;
import ci.gouv.dgbf.extension.server.service.api.request.FilterDto;
import lombok.Getter;
import lombok.Setter;

/**
 * Cette classe représente le filtre de {@link PayrollComponentProjectionDto}.
 *
 * @author Christian
 */
@Getter
@Setter
public class PayrollComponentProjectionFilter extends AbstractIdentifiableFilter {
  /**
   * Identifiant de la periode.
   */
  String payrollPeriodIdentifier;
  /**
   * Identifiant de l'employé.
   */
  String employeeIdentifier;
  /**
   * Identifiant de l'exercice.
   */
  String exerciseIdentifier;

  /**
   * Identifiant de l'unité administrative.
   */
  String administrativeUnitIdentifier;

  /**
   * Cette méthode permet d'instantier un objet.
   *
   * @param dto filtre
   */
  public PayrollComponentProjectionFilter(FilterDto dto) {
    super(dto);
  }

  public PayrollComponentProjectionFilter() {
  }

  @Override
  protected void doInitialize(FilterDto filter) {
    super.doInitialize(filter);
    payrollPeriodIdentifier = getPayrollPeriodIdentifier(filter);
    employeeIdentifier = getEmployeeIdentifier(filter);
    exerciseIdentifier = getExerciseIdentifier(filter);
    administrativeUnitIdentifier = getAdministrativeUnitIdentifier(filter);
  }

  @Override
  protected void toDto(FilterDto filter) {
    super.toDto(filter);

    setPayrollPeriodIdentifier(filter, payrollPeriodIdentifier);
    setEmployeeIdentifier(filter, employeeIdentifier);
    setExerciseIdentifier(filter, exerciseIdentifier);
    setAdministrativeUnitIdentifier(filter, administrativeUnitIdentifier);
  }

  /**
   * Cette méthode permet d'obtenir l'identifiant de l'unité administrative.
   *
   * @param filter filtre
   * @return identifiant  de l'unité administrative
   */
  public static String getAdministrativeUnitIdentifier(FilterDto filter) {
    return get(filter, d -> d.getFieldValueAsStringByName(JSON_ADMINISTRATIVE_UNIT_IDENTIFIER));
  }

  /**
   * Cette méthode permet d'assigner l'identifiant de l'unité administrative.
   *
   * @param filter     filtre
   * @param identifier identifiant de l'unité administrative
   */
  public static void setAdministrativeUnitIdentifier(FilterDto filter, String identifier) {
    set(filter, JSON_ADMINISTRATIVE_UNIT_IDENTIFIER, f -> f.getValueAsString(),
        f -> f.setValueAsString(identifier));
  }

  /**
   * Cette méthode permet d'obtenir l'identifiant de la periode.
   *
   * @param filter filtre
   * @return identifiant  de la periode
   */
  public static String getPayrollPeriodIdentifier(FilterDto filter) {
    return get(filter, d -> d.getFieldValueAsStringByName(JSON_PAYROLL_PERIOD_IDENTIFIER));
  }

  /**
   * Cette méthode permet d'assigner l'identifiant de la periode.
   *
   * @param filter     filtre
   * @param identifier identifiant de l'identifiant de la periode
   */
  public static void setPayrollPeriodIdentifier(FilterDto filter, String identifier) {
    set(filter, JSON_PAYROLL_PERIOD_IDENTIFIER, f -> f.getValueAsString(),
        f -> f.setValueAsString(identifier));
  }

  /**
   * Cette méthode permet d'obtenir l'identifiant de l'exercice.
   *
   * @param filter filtre
   * @return identifiant  de l'exercice
   */
  public static String getEmployeeIdentifier(FilterDto filter) {
    return get(filter, d -> d.getFieldValueAsStringByName(JSON_EXERCISE_IDENTIFIER));
  }

  /**
   * Cette méthode permet d'assigner l'identifiant de l'exercice.
   *
   * @param filter     filtre
   * @param identifier identifiant de l'identifiant de l'exercice
   */
  public static void setEmployeeIdentifier(FilterDto filter, String identifier) {
    set(filter, JSON_EXERCISE_IDENTIFIER, f -> f.getValueAsString(),
        f -> f.setValueAsString(identifier));
  }

  /**
   * Cette méthode permet d'obtenir l'identifiant de l'employé.
   *
   * @param filter filtre
   * @return identifiant  de  l'employé
   */
  public static String getExerciseIdentifier(FilterDto filter) {
    return get(filter, d -> d.getFieldValueAsStringByName(JSON_EXERCISE_IDENTIFIER));
  }

  /**
   * Cette méthode permet d'assigner l'identifiant de l'employé.
   *
   * @param filter     filtre
   * @param identifier identifiant de l'identifiant de l'employé
   */
  public static void setExerciseIdentifier(FilterDto filter, String identifier) {
    set(filter, JSON_EXERCISE_IDENTIFIER, f -> f.getValueAsString(),
        f -> f.setValueAsString(identifier));
  }

  /**
   * Identifiant json champ exrecice.
   */
  public static final String JSON_EXERCISE_IDENTIFIER = "idExercice";

  /**
   * Identifiant json unité Administrative.
   */
  public static final String JSON_ADMINISTRATIVE_UNIT_IDENTIFIER = "idUniteAdministrative";

  /**
   * Identifiant json employé.
   */
  public static final String JSON_EMPLOYEE_IDENTIFIER = "idEmploye";


  /**
   * Identifiant json champ periode.
   */
  public static final String JSON_PAYROLL_PERIOD_IDENTIFIER = "idPeriodePaie";
}
