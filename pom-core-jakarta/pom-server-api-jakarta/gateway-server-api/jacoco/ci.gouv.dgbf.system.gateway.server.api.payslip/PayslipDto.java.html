package ci.gouv.dgbf.system.gateway.server.api.payslip;

import ci.gouv.dgbf.extension.server.service.api.entity.AbstractIdentifiableCodableNamableDto;
import jakarta.json.bind.annotation.JsonbProperty;
import lombok.EqualsAndHashCode;
import lombok.Getter;
import lombok.Setter;

/**
 * Cette classe représente un bulletin de salaire.
 *
 * @author Christian
 *
 */
@Getter
@Setter
@EqualsAndHashCode(callSuper = true)
public class PayslipDto extends AbstractIdentifiableCodableNamableDto {

  /**
   * Identifiant du fichier.
   */
  @JsonbProperty(value = JSON_FILE_IDENTIFIER)
  private String fileIdentifier;

  /**
   * Représentation en chaine de caractères de {@link PeriodDto}.
   */
  @JsonbProperty(value = JSON_PERIOD_AS_STRING)
  private String periodAsString;
  
  /**
   * Année.
   */
  @JsonbProperty(value = JSON_PERIOD_YEAR)
  private Integer year;
  
  /**
   * Mois.
   */
  @JsonbProperty(value = JSON_PERIOD_MONTH)
  private Integer month;
  
  /**
   * Identifiant JSON {@link #fileIdentifier}.
   */
  public static final String JSON_FILE_IDENTIFIER = "idFichier";
  
  /**
   * Identifiant JSON {@link #periodAsString}.
   */
  public static final String JSON_PERIOD_AS_STRING = "periodeChaine";
  
  /**
   * Identifiant JSON {@link #year}.
   */
  public static final String JSON_PERIOD_YEAR = "annee";
  
  /**
   * Identifiant JSON {@link #month}.
   */
  public static final String JSON_PERIOD_MONTH = "mois";
  
  /**
   * Nom pour lecture humaine de la classe.
   */
  public static final String NAME = "bulletin de salaire";

  /**
   * Nom au pluriel pour lecture humaine de la classe.
   */
  public static final String PLURAL_NAME = "bulletins de salaire";
}
