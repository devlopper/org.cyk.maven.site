package ci.gouv.dgbf.system.gateway.server.api.file;

import ci.gouv.dgbf.extension.server.service.api.entity.AbstractIdentifiableNamableDto;
import jakarta.json.bind.annotation.JsonbProperty;
import lombok.EqualsAndHashCode;
import lombok.Getter;
import lombok.Setter;

/**
 * Cette classe représente un fichier.
 *
 * @author Christian Yao Komenan
 *
 */
@Getter
@Setter
@EqualsAndHashCode(callSuper = true)
public class FileDto extends AbstractIdentifiableNamableDto {

  /**
   * Type de contenu.
   */
  @JsonbProperty(value = JSON_CONTENT_TYPE)
  private String contentType;

  /**
   * Octets.
   */
  @JsonbProperty(value = JSON_BYTES)
  private byte[] bytes;
  
  /**
   * Nombre d'octets.
   */
  @JsonbProperty(value = JSON_NUMBER_OF_BYTES)
  private Integer numberOfBytes;
  
  /**
   * Représentation en chaine de caractères de {@link #numberOfBytes}.
   */
  @JsonbProperty(value = JSON_NUMBER_OF_BYTES_AS_STRING)
  private String numberOfBytesAsString;
  
  /**
   * Identifiant {@link OwnerDto}.
   */
  @JsonbProperty(value = JSON_OWNER_IDENTIFIER)
  private String ownerIdentifier;

  /**
   * Représentation en chaine de caratères de {@link OwnerDto}.
   */
  @JsonbProperty(value = JSON_OWNER_AS_STRING)
  private String ownerAsString;
  
  /**
   * Hash.
   */
  @JsonbProperty(value = JSON_HASH)
  private String hash;

  /**
   * Identifiant {@link FilesGroupDto}.
   */
  @JsonbProperty(value = JSON_GROUP_IDENTIFIER)
  private String groupIdentifier;
  
  /**
   * Identifiant JSON {@link #contentType}.
   */
  public static final String JSON_CONTENT_TYPE = "typeContenu";

  /**
   * Identifiant JSON {@link #bytes}.
   */
  public static final String JSON_BYTES = "octets";
  
  /**
   * Identifiant JSON {@link #numberOfBytes}.
   */
  public static final String JSON_NUMBER_OF_BYTES = "nombreOctets";
  
  /**
   * Identifiant JSON {@link #numberOfBytesAsString}.
   */
  public static final String JSON_NUMBER_OF_BYTES_AS_STRING = "nombreOctetsChaine";
  
  /**
   * Identifiant JSON {@link #ownerIdentifier}.
   */
  public static final String JSON_OWNER_IDENTIFIER = "idProprietaire";
  
  /**
   * Identifiant JSON {@link #ownerAsString}.
   */
  public static final String JSON_OWNER_AS_STRING = "proprietaireChaine";
  
  /**
   * Identifiant JSON {@link #groupIdentifier}.
   */
  public static final String JSON_GROUP_IDENTIFIER = "idGroupe";
  
  /**
   * Identifiant JSON {@link #hash}.
   */
  public static final String JSON_HASH = "hash";
  
  /**
   * Nom pour lecture humaine de la classe.
   */
  public static final String NAME = "fichier";
  
  /**
   * Nom au pluriel pour lecture humaine de la classe.
   */
  public static final String PLURAL_NAME = NAME + "s";
}
