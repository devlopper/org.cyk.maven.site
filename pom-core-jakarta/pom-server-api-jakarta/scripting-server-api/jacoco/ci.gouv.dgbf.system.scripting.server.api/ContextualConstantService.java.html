/**

@Path(value = ContextualConstantService.PATH)
@Tag(name = "Gestion des constantes contextuelles")
public interface ContextualConstantService extends SpecificService {

  /**
   * Chemin de base des services.
   */
  String PATH = "constantes-contextuelles";

  /**
   * Cette interface représente la requête d'enregistrement.
   *
   * @author Christian
   */
  interface ContextualConstantSaveRequestDto {
    /**
     * Cette méthode permet d'obtenir l'identifiant de la constante.
     *
     * @return l'identifiant de la constante
     */
    String getConstantIdentifier();

    /**
     * Cette méthode permet d'assigner l'identifiant de la constante.
     *
     * @param constantIdentifier l'identifiant de la constante
     */
    void setConstantIdentifier(String constantIdentifier);

    /**
     * Cette méthode permet d'obtenir l'identifiant du groupe.
     *
     * @return l'identifiant du groupe
     */
    String getGroupIdentifier();

    /**
     * Cette méthode permet d'assigner l'identifiant du groupe.
     *
     * @param groupIdentifier l'identifiant du groupe
     */
    void setGroupIdentifier(String groupIdentifier);

    /**
     * Cette méthode permet d'obtenir l'identifiant de la catégorie.
     *
     * @return l'identifiant de la catégorie
     */
    String getCategoryIdentifier();

    /**
     * Cette méthode permet d'assigner l'identifiant de la catégorie.
     *
     * @param categoryIdentifier l'identifiant de la catégorie
     */
    void setCategoryIdentifier(String categoryIdentifier);

    /**
     * Cette méthode permet d'obtenir la date de début.
     *
     * @return la date de début
     */
    LocalDateTime getFromDate();

    /**
     * Cette méthode permet d'assigner la date de début.
     *
     * @param fromDate la date de début
     */
    void setFromDate(LocalDateTime fromDate);

    /**
     * Cette méthode permet d'obtenir la date de fin.
     *
     * @return la date de fin
     */
    LocalDateTime getToDate();

    /**
     * Cette méthode permet d'assigner la date de fin.
     *
     * @param toDate la date de fin
     */
    void setToDate(LocalDateTime toDate);

    /**
     * {@link HasConstantIdentifierDto#JSON_CONSTANT_IDENTIFIER}.
     */
    String JSON_CONSTANT_IDENTIFIER = HasConstantIdentifierDto.JSON_CONSTANT_IDENTIFIER;

    /**
     * {@link HasGroupIdentifierDto#JSON_GROUP_IDENTIFIER}.
     */
    String JSON_GROUP_IDENTIFIER = HasGroupIdentifierDto.JSON_GROUP_IDENTIFIER;

    /**
     * {@link HasCategoryIdentifierDto#JSON_CATEGORY_IDENTIFIER}.
     */
    String JSON_CATEGORY_IDENTIFIER = HasCategoryIdentifierDto.JSON_CATEGORY_IDENTIFIER;

    /**
     * {@link HasFromDateDto#JSON_FROM_DATE}.
     */
    String JSON_FROM_DATE = HasFromDateDto.JSON_FROM_DATE;

    /**
     * {@link HasToDateDto#JSON_TO_DATE}.
     */
    String JSON_TO_DATE = HasToDateDto.JSON_TO_DATE;
  }

  /**
   * Identifiant du service de création.
   */
  String CREATE_IDENTIFIER = "CREATION_CONSTANTE_CONTEXTUELLE";

  /**
   * Chemin du service de création.
   */
  String CREATE_PATH = "";

  /**
   * Cette méthode permet de créer {@link ConstantDto}.

  @POST
  @Consumes(MediaType.APPLICATION_JSON)
  @Produces(MediaType.APPLICATION_JSON)
  @Operation(operationId = CREATE_IDENTIFIER)
  @APIResponse(responseCode = "201",
      content = {@Content(schema = @Schema(implementation = CreateResponseDto.class))})
  Response create(ContextualConstantCreateRequestDto request);

    private String categoryIdentifier;

    @JsonbProperty(JSON_FROM_DATE)
    private LocalDateTime fromDate;

    @JsonbProperty(JSON_TO_DATE)
    private LocalDateTime toDate;
  }
