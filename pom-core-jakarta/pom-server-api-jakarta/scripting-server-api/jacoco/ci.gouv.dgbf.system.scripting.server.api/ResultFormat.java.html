package ci.gouv.dgbf.system.scripting.server.api;


import java.util.stream.Stream;
import lombok.Getter;

/**
 * Cette classe représente les formats de résultat de {@link ScriptDto}.
 *
 * @author Christian
 *
 */
@Getter
@Deprecated(forRemoval = true)
public enum ResultFormat {

  /**
   * Javascript Object Notation.
   */
  JSON(ResultFormatCode.JSON, "Javascript Object Notation"),

  /**
   * Entier.
   */
  NONE(ResultFormatCode.NONE, "Aucun");

  private ResultFormat(String code, String name) {
    this.code = code;
    this.name = name;
  }

  private String code;
  private String name;

  @Override
  public String toString() {
    return name;
  }

  /**
   * Cette méthode permet d'obtenir {@link ResultFormat} par code.
   *
   * @param code code
   * @return statut
   */
  public static ResultFormat getByCode(String code) {
    return Stream.of(ResultFormat.values()).filter(status -> status.getCode().equals(code))
        .findFirst().orElseThrow(IllegalArgumentException::new);
  }
  
  /**
   * Identifiant json de l'identifiant.
   */
  public static final String JSON_THIS = "formatResultat";
  
  /**
   * Identifiant json de la représentation en chaine de caractères.
   */
  public static final String JSON_THIS_AS_STRING = "formatResultatChaine";
}
