package ci.gouv.dgbf.system.scripting.server.api;


import java.util.stream.Stream;
import lombok.Getter;

/**
 * Cette classe représente les types de résultat de {@link ScriptDto}.
 *
 * @author Christian
 *
 */
@Getter
@Deprecated
public enum ResultType {

  /**
   * Chaine de caractères.
   */
  STRING(ResultTypeCode.STRING, "Chaine de caractères"),

  /**
   * Entier.
   */
  INTEGER(ResultTypeCode.INTEGER, "Entier"),

  /**
   * Booléen.
   */
  BOOLEAN(ResultTypeCode.BOOLEAN, "Booléen"),

  /**
   * Carte.
   */
  MAP(ResultTypeCode.MAP, "Carte");

  private ResultType(String code, String name) {
    this.code = code;
    this.name = name;
  }

  private String code;
  private String name;

  @Override
  public String toString() {
    return name;
  }

  /**
   * Cette méthode permet d'obtenir {@link ResultType} par code.
   *
   * @param code code
   * @return statut
   */
  public static ResultType getByCode(String code) {
    return Stream.of(ResultType.values()).filter(status -> status.getCode().equals(code))
        .findFirst().orElseThrow(IllegalArgumentException::new);
  }

  /**
   * Cette méthode permet parser une chaine de caractères afin d'obtenir une valeur du type
   * souhaité.
   *
   * @param string chaine de caractères
   * @param type {@link ResultType}
   * @return valeur du type souhaité
   */
  public static Object parse(String string, ResultType type) {
    if (ResultType.INTEGER.equals(type)) {
      return Integer.parseInt(string);
    }
    if (ResultType.BOOLEAN.equals(type)) {
      return Boolean.parseBoolean(string);
    }
    return string;
  }

  /**
   * Identifiant json de l'identifiant.
   */
  public static final String JSON_THIS = "typeResultat";

  /**
   * Identifiant json de la représentation en chaine de caractères.
   */
  public static final String JSON_THIS_AS_STRING = "typeResultatChaine";
}
