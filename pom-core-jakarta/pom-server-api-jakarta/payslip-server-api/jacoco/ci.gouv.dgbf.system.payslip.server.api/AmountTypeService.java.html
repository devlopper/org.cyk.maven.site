package ci.gouv.dgbf.system.payslip.server.api;

import ci.gouv.dgbf.extension.server.service.api.SpecificService;
import ci.gouv.dgbf.extension.server.service.api.request.AbstractAuditedRequestJsonDto;
import ci.gouv.dgbf.extension.server.service.api.request.AbstractRequestDto;
import jakarta.json.bind.annotation.JsonbPropertyOrder;
import jakarta.ws.rs.Consumes;
import jakarta.ws.rs.POST;
import jakarta.ws.rs.Path;
import jakarta.ws.rs.Produces;
import jakarta.ws.rs.core.MediaType;
import jakarta.ws.rs.core.Response;
import lombok.Getter;
import lombok.Setter;
import lombok.experimental.Accessors;
import org.eclipse.microprofile.openapi.annotations.Operation;
import org.eclipse.microprofile.openapi.annotations.media.Content;
import org.eclipse.microprofile.openapi.annotations.media.Schema;
import org.eclipse.microprofile.openapi.annotations.responses.APIResponse;
import org.eclipse.microprofile.openapi.annotations.tags.Tag;

/**
 * Cette interface représente les services concernant les types de montants.
 *
 * @author Christian
 *
 */
@Path(AmountTypeService.PATH)
@Tag(name = "Gestion des types de montant")
public interface AmountTypeService extends SpecificService {

  /**
   * Chemin de base.
   */
  String PATH = "types-montant";

  /**
   * Identifiant du service d'obtention de l'identifiant de bulletin de salaire.
   */
  String GET_PAYSLIP_IDENTIFIER_IDENTIFIER = "OBTENTION_IDENTIFIANT_BULLETIN_SALAIRE";

  /**
   * Chemin du service d'obtention de l'identifiant de bulletin de salaire.
   */
  String GET_PAYSLIP_IDENTIFIER_PATH = "identifiant-bulletin-salaire";

  /**
   * Cette méthode permet d'obtenir l'identifiant de bulletin de salaire.
   *
   * @param request requête
   * @return réponse
   */
  @Path(GET_PAYSLIP_IDENTIFIER_PATH)
  @POST
  @Consumes(MediaType.APPLICATION_JSON)
  @Produces(MediaType.TEXT_PLAIN)
  @Operation(operationId = GET_PAYSLIP_IDENTIFIER_IDENTIFIER,
      summary = "Obtention de l'identifiant de bulletin de salaire",
      description = "Ce service permet d'obtenir l'identifiant de bulletin de salaire")
  @APIResponse(responseCode = "200",
      content = {@Content(schema = @Schema(implementation = String.class))})
  Response getPayslipIdentifier(GetPayslipIdentifierRequestDto request);

  /**
   * Cette classe représente l'objet de transfert de donnée de la requête d'obtention de fonction
   * publique.
   *
   * @author Christian
   *
   */
  @Getter
  @Setter
  @Accessors(chain = true)
  @JsonbPropertyOrder(value = {AbstractRequestDto.FIELD_AUDIT_WHO,
      AbstractAuditedRequestJsonDto.FIELD_AUDIT_SESSION})
  class GetPayslipIdentifierRequestDto extends AbstractAuditedRequestJsonDto {

  }

  /**
   * Identifiant du service d'obtention de l'identifiant de arriéré de salaire.
   */
  String GET_ARREARAGE_IDENTIFIER_IDENTIFIER = "OBTENTION_IDENTIFIANT_ARRIERE_SALAIRE";

  /**
   * Chemin du service d'obtention de l'identifiant de arriéré de salaire.
   */
  String GET_ARREARAGE_IDENTIFIER_PATH = "identifiant-arriere-salaire";

  /**
   * Cette méthode permet d'obtenir l'identifiant de arriéré de salaire.
   *
   * @param request requête
   * @return réponse
   */
  @Path(GET_ARREARAGE_IDENTIFIER_PATH)
  @POST
  @Consumes(MediaType.APPLICATION_JSON)
  @Produces(MediaType.TEXT_PLAIN)
  @Operation(operationId = GET_ARREARAGE_IDENTIFIER_IDENTIFIER,
      summary = "Obtention de l'identifiant de arriéré de salaire",
      description = "Ce service permet d'obtenir l'identifiant de arriéré de salaire")
  @APIResponse(responseCode = "200",
      content = {@Content(schema = @Schema(implementation = String.class))})
  Response getArrearageIdentifier(GetArrearageIdentifierRequestDto request);

  /**
   * Cette classe représente l'objet de transfert de donnée de la requête d'obtention de fonction
   * publique.
   *
   * @author Christian
   *
   */
  @Getter
  @Setter
  @Accessors(chain = true)
  @JsonbPropertyOrder(value = {AbstractRequestDto.FIELD_AUDIT_WHO,
      AbstractAuditedRequestJsonDto.FIELD_AUDIT_SESSION})
  class GetArrearageIdentifierRequestDto extends AbstractAuditedRequestJsonDto {

  }
}
