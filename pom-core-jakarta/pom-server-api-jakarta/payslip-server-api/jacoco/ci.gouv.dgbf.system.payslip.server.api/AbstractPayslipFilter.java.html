package ci.gouv.dgbf.system.payslip.server.api;

import ci.gouv.dgbf.extension.server.service.api.AbstractIdentifiableFilter;
import ci.gouv.dgbf.extension.server.service.api.request.FilterDto;
import lombok.Getter;
import lombok.Setter;

/**
 * Cette classe représente la base de filtre de bulletin de salaire.
 *
 * @author Christian
 *
 */
@Getter
@Setter
public abstract class AbstractPayslipFilter extends AbstractIdentifiableFilter {

  /**
   * Identifiant de l'employé.
   */
  String employeeIdentifier;

  /**
   * Matricule fonctionnaire.
   */
  String functionaryRegistrationNumber;

  /**
   * Identifiant {@link PeriodDto}.
   */
  String periodIdentifier;

  /**
   * Nom utilisateur.
   */
  String userName;

  /**
   * Cette méthode permet de construire.
   *
   * @param dto dto
   */
  protected AbstractPayslipFilter(FilterDto dto) {
    super(dto);
  }

  /**
   * Cette méthode permet de construire.
   */
  protected AbstractPayslipFilter() {}

  @Override
  protected void doInitialize(FilterDto filter) {
    super.doInitialize(filter);
    employeeIdentifier = getEmployeeIdentifier(filter);
    functionaryRegistrationNumber = getFunctionaryRegistrationNumber(filter);
    userName = getUserName(filter);
    periodIdentifier = getPeriodIdentifier(filter);
  }

  @Override
  protected void toDto(FilterDto filter) {
    super.toDto(filter);
    setEmployeeIdentifier(filter, employeeIdentifier);
    setFunctionaryRegistrationNumber(filter, functionaryRegistrationNumber);
    setUserName(filter, userName);
    setPeriodIdentifier(filter, periodIdentifier);
  }

  /**
   * Cette méthode permet d'assigner l'identifiant de l'employé.
   *
   * @param filter filtre
   * @param identifier identifiant de l'employé.
   */
  public static void setEmployeeIdentifier(FilterDto filter, String identifier) {
    set(filter, JSON_EMPLOYEE_IDENTIFIER, f -> f.getValueAsString(),
        f -> f.setValueAsString(identifier));
  }

  /**
   * Cette méthode permet d'obtenir l'identifiant de l'employé.
   *
   * @param filter filtre
   * @return identifiant de l'employé.
   */
  public static String getEmployeeIdentifier(FilterDto filter) {
    return get(filter, d -> d.getFieldValueAsStringByName(JSON_EMPLOYEE_IDENTIFIER));
  }

  /**
   * Cette méthode permet d'assigner le matricule fonctionnaire.
   *
   * @param filter filtre
   * @param functionaryRegistrationNumber matricule fonctionnaire.
   */
  public static void setFunctionaryRegistrationNumber(FilterDto filter,
      String functionaryRegistrationNumber) {
    set(filter, JSON_FUNCTIONARY_REGISTRATION_NUMBER, f -> f.getValueAsString(),
        f -> f.setValueAsString(functionaryRegistrationNumber));
  }

  /**
   * Cette méthode permet d'obtenir la matricule fonctionnaire.
   *
   * @param filter filtre
   * @return matricule fonctionnaire.
   */
  public static String getFunctionaryRegistrationNumber(FilterDto filter) {
    return get(filter, d -> d.getFieldValueAsStringByName(JSON_FUNCTIONARY_REGISTRATION_NUMBER));
  }

  /**
   * Cette méthode permet d'assigner l'identifiant de {@link PeriodDto}.
   *
   * @param filter filtre
   * @param identifier identifiant de {@link PeriodDto}.
   */
  public static void setPeriodIdentifier(FilterDto filter, String identifier) {
    set(filter, JSON_PERIOD_IDENTIFIER, f -> f.getValueAsString(),
        f -> f.setValueAsString(identifier));
  }

  /**
   * Cette méthode permet d'obtenir l'identifiant de {@link PeriodDto}.
   *
   * @param filter filtre
   * @return identifiant de {@link PeriodDto}.
   */
  public static String getPeriodIdentifier(FilterDto filter) {
    return get(filter, d -> d.getFieldValueAsStringByName(JSON_PERIOD_IDENTIFIER));
  }

  /**
   * Cette méthode permet d'assigner le nom utilisateur.
   *
   * @param filter filtre
   * @param userName nom utilisateur.
   */
  public static void setUserName(FilterDto filter, String userName) {
    set(filter, JSON_USER_NAME, f -> f.getValueAsString(), f -> f.setValueAsString(userName));
  }

  /**
   * Cette méthode permet d'obtenir le nom utilisateur.
   *
   * @param filter filtre
   * @return userName nom utilisateur.
   */
  public static String getUserName(FilterDto filter) {
    return get(filter, d -> d.getFieldValueAsStringByName(JSON_USER_NAME));
  }

  /**
   * {@link AbstractEmployeePeriodDto#JSON_EMPLOYEE_IDENTIFIER}.
   */
  public static final String JSON_EMPLOYEE_IDENTIFIER =
      AbstractEmployeePeriodDto.JSON_EMPLOYEE_IDENTIFIER;

  /**
   * {@link AbstractEmployeePeriodDto#JSON_FUNCTIONARY_REGISTRATION_NUMBER}.
   */
  public static final String JSON_FUNCTIONARY_REGISTRATION_NUMBER =
      AbstractEmployeePeriodDto.JSON_FUNCTIONARY_REGISTRATION_NUMBER;

  /**
   * {@link AbstractEmployeePeriodDto#JSON_PERIOD_IDENTIFIER}.
   */
  public static final String JSON_PERIOD_IDENTIFIER =
      AbstractEmployeePeriodDto.JSON_PERIOD_IDENTIFIER;

  /**
   * Identifiant JSON {@link #userName}.
   */
  public static final String JSON_USER_NAME = "nomUtilisateur";
}
