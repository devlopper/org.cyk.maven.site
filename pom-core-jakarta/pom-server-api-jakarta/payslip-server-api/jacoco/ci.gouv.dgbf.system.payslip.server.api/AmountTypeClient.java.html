package ci.gouv.dgbf.system.payslip.server.api;

import ci.gouv.dgbf.extension.server.service.api.client.AbstractClient;
import ci.gouv.dgbf.extension.server.service.api.client.executor.GetStringExecutor;
import ci.gouv.dgbf.system.payslip.server.api.AmountTypeService.GetArrearageIdentifierRequestDto;
import ci.gouv.dgbf.system.payslip.server.api.AmountTypeService.GetPayslipIdentifierRequestDto;
import jakarta.enterprise.context.ApplicationScoped;

/**
 * Cette classe représente le client de GenderService.
 *
 * @author Christian
 *
 */
@ApplicationScoped
public class AmountTypeClient extends AbstractClient<AmountTypeService> {

  @Override
  public AmountTypeClient service(AmountTypeService service) {
    return (AmountTypeClient) super.service(service);
  }

  /**
   * Cette méthode permet d'obtenir l'identifiant de bulletin de salaire.
   *
   * @param request requête
   * @return réponse
   */
  public String getPayslipIdentifier(GetPayslipIdentifierRequestDto request) {
    return new GetStringExecutor(
        AmountTypeService.GET_PAYSLIP_IDENTIFIER_IDENTIFIER)
            .execute(() -> service().getPayslipIdentifier(request));
  }

  /**
   * Cette méthode permet d'obtenir l'identifiant de bulletin de salaire.
   *
   * @param auditWho audit acteur
   * @param auditSession audit session
   * @return identifiant bulletin de salaire
   */
  public String getPayslipIdentifier(String auditWho, String auditSession) {
    GetPayslipIdentifierRequestDto request = new GetPayslipIdentifierRequestDto();
    request.setAuditWho(auditWho);
    request.setAuditSession(auditSession);
    return getPayslipIdentifier(request);
  }
  
  /**
   * Cette méthode permet d'obtenir l'identifiant de arriéré de salaire.
   *
   * @param request requête
   * @return réponse
   */
  public String getArrearageIdentifier(GetArrearageIdentifierRequestDto request) {
    return new GetStringExecutor(
        AmountTypeService.GET_PAYSLIP_IDENTIFIER_IDENTIFIER)
            .execute(() -> service().getArrearageIdentifier(request));
  }

  /**
   * Cette méthode permet d'obtenir l'identifiant de arriéré de salaire.
   *
   * @param auditWho audit acteur
   * @param auditSession audit session
   * @return identifiant arriéré de salaire
   */
  public String getArrearageIdentifier(String auditWho, String auditSession) {
    GetArrearageIdentifierRequestDto request = new GetArrearageIdentifierRequestDto();
    request.setAuditWho(auditWho);
    request.setAuditSession(auditSession);
    return getArrearageIdentifier(request);
  }
}
