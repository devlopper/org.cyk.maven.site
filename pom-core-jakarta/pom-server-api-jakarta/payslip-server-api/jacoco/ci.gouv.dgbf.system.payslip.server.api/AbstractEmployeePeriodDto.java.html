package ci.gouv.dgbf.system.payslip.server.api;

import ci.gouv.dgbf.extension.server.service.api.entity.AbstractIdentifiableCodableAuditableDto;
import jakarta.json.bind.annotation.JsonbProperty;
import lombok.EqualsAndHashCode;
import lombok.Getter;
import lombok.Setter;

/**
 * Cette classe représente une période d'un employé.
 *
 * @author Christian
 *
 */
@Getter
@Setter
@EqualsAndHashCode(callSuper = true)
public abstract class AbstractEmployeePeriodDto extends AbstractIdentifiableCodableAuditableDto {

  /**
   * Identifiant de l'employé.
   */
  @JsonbProperty(value = JSON_EMPLOYEE_IDENTIFIER)
  protected String employeeIdentifier;

  /**
   * Représentation en chaine de caractères de employé.
   */
  @JsonbProperty(value = JSON_EMPLOYEE_AS_STRING)
  protected String employeeAsString;
  
  /**
   * Matricule de l'employé.
   */
  @JsonbProperty(value = JSON_EMPLOYEE_REGISTRATION_NUMBER)
  protected String employeeRegistrationNumber;
  
  /**
   * Nom et prénoms de l'employé.
   */
  @JsonbProperty(value = JSON_EMPLOYEE_NAMES)
  protected String employeeNames;
  
  /**
   * Adresse email de l'employé.
   */
  @JsonbProperty(value = JSON_EMPLOYEE_EMAIL_ADDRESS)
  protected String employeeEmailAddress;
  
  /**
   * Identifiant de {@link PeriodDto}.
   */
  @JsonbProperty(value = JSON_PERIOD_IDENTIFIER)
  protected String periodIdentifier;

  /**
   * Représentation en chaine de caractères de {@link PeriodDto}.
   */
  @JsonbProperty(value = JSON_PERIOD_AS_STRING)
  protected String periodAsString;
  
  /**
   * Année.
   */
  @JsonbProperty(value = JSON_PERIOD_YEAR)
  protected Integer year;
  
  /**
   * Mois.
   */
  @JsonbProperty(value = JSON_PERIOD_MONTH)
  protected Integer month;
  
  /**
   * Identifiant JSON {@link #employeeIdentifier}.
   */
  public static final String JSON_EMPLOYEE_IDENTIFIER = "idEmploye";
  
  /**
   * Identifiant JSON {@link #employeeAsString}.
   */
  public static final String JSON_EMPLOYEE_AS_STRING = "employeChaine";
  
  /**
   * Identifiant JSON {@link #employeeRegistrationNumber}.
   */
  public static final String JSON_EMPLOYEE_REGISTRATION_NUMBER = "matricule";
  
  /**
   * Identifiant JSON {@link #employeeNames}.
   */
  public static final String JSON_EMPLOYEE_NAMES = "nomEtPrenoms";
  
  /**
   * Identifiant JSON {@link #employeeEmailAddress}.
   */
  public static final String JSON_EMPLOYEE_EMAIL_ADDRESS = "adresseEmail";
  
  /**
   * Identifiant JSON {@link #periodIdentifier}.
   */
  public static final String JSON_PERIOD_IDENTIFIER = "idPeriode";
  
  /**
   * Identifiant JSON {@link #periodAsString}.
   */
  public static final String JSON_PERIOD_AS_STRING = "periodeChaine";
  
  /**
   * Identifiant JSON {@link #year}.
   */
  public static final String JSON_PERIOD_YEAR = "annee";
  
  /**
   * Identifiant JSON {@link #month}.
   */
  public static final String JSON_PERIOD_MONTH = "mois";
}
