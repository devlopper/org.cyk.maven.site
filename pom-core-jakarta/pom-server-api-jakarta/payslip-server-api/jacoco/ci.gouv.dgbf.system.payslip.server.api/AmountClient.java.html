package ci.gouv.dgbf.system.payslip.server.api;

import ci.gouv.dgbf.extension.server.service.api.client.AbstractClient;
import ci.gouv.dgbf.extension.server.service.api.client.executor.GetOneExecutor;
import ci.gouv.dgbf.system.payslip.server.api.AmountService.LoadByYearByMonthByTypeIdentifierRequestDto;
import ci.gouv.dgbf.system.payslip.server.api.AmountService.LoadByYearByMonthByTypeIdentifierResponseDto;
import jakarta.enterprise.context.ApplicationScoped;
import lombok.Setter;
import lombok.experimental.Accessors;

/**
 * Cette classe représente le client de {@link AmountService}.
 *
 * @author Christian
 *
 */
@ApplicationScoped
@Setter
@Accessors(chain = true, fluent = true)
public class AmountClient extends AbstractClient<AmountService> {

  @Override
  public AmountClient service(AmountService service) {
    return (AmountClient) super.service(service);
  }

  /**
   * Cette méthode permet de charger les arriérés par année, mois et type de montant.
   *
   * @param request requête
   * @return réponse
   */
  public LoadByYearByMonthByTypeIdentifierResponseDto loadByYearByMonthByTypeIdentifier(
      LoadByYearByMonthByTypeIdentifierRequestDto request) {
    return new GetOneExecutor<LoadByYearByMonthByTypeIdentifierResponseDto>(
        LoadByYearByMonthByTypeIdentifierResponseDto.class,
        AmountService.LOAD_BY_YEAR_BY_MONTH_BY_TYPE_IDENTIFIER_IDENTIFIER)
            .execute(() -> service().loadByYearByMonthByTypeIdentifier(request));
  }

  /**
   * Cette méthode permet de charger les arriérés par année, mois et type de montant.
   *
   * @param year année
   * @param month mois
   * @param typeIdentifier identifiant du type de montant
   * @param registrationNumberGroupIdentifier identifiant du groupe de matricule
   * @param auditWho audit acteur
   * @param auditSession audit session
   * @return montants
   */
  public LoadByYearByMonthByTypeIdentifierResponseDto loadByYearByMonthByTypeIdentifier(
      Integer year, Integer month, String typeIdentifier, String registrationNumberGroupIdentifier,
      String auditWho, String auditSession) {
    LoadByYearByMonthByTypeIdentifierRequestDto request =
        new LoadByYearByMonthByTypeIdentifierRequestDto();
    request.setYear(year);
    request.setMonth(month);
    request.setTypeIdentifier(typeIdentifier);
    request.setRegistrationNumberGroupIdentifier(registrationNumberGroupIdentifier);
    request.setAuditWho(auditWho);
    request.setAuditSession(auditSession);
    return loadByYearByMonthByTypeIdentifier(request);
  }
}
