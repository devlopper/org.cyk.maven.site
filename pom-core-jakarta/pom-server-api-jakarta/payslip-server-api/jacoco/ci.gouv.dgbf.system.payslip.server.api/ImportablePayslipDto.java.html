package ci.gouv.dgbf.system.payslip.server.api;

import ci.gouv.dgbf.extension.server.service.api.entity.AbstractIdentifiableAuditableDto;
import jakarta.json.bind.annotation.JsonbProperty;
import lombok.EqualsAndHashCode;
import lombok.Getter;
import lombok.Setter;

/**
 * Cette classe représente un {@link PayslipDto} importable.
 *
 * @author Christian
 *
 */
@Getter
@Setter
@EqualsAndHashCode(callSuper = true)
public class ImportablePayslipDto extends AbstractIdentifiableAuditableDto {
  
  /**
   * Représentation en chaine de caractères de {@link EmployeeDto}.
   */
  @JsonbProperty(value = JSON_EMPLOYEE_AS_STRING)
  private String employeeAsString;

  /**
   * Matricule de {@link EmployeeDto}.
   */
  @JsonbProperty(value = JSON_REGISTRATION_NUMBER)
  private String registrationNumber;
  
  /**
   * Représentation en chaine de caractères de {@link PeriodDto}.
   */
  @JsonbProperty(value = JSON_PERIOD_AS_STRING)
  private String periodAsString;
  
  /**
   * Nom d'utilisateur.
   */
  @JsonbProperty(value = JSON_USER_NAME)
  private String userName;
  
  /**
   * Identifiant JSON {@link #employeeAsString}.
   */
  public static final String JSON_EMPLOYEE_AS_STRING = "employeChaine";
  
  /**
   * Identifiant JSON {@link #registrationNumber}.
   */
  public static final String JSON_REGISTRATION_NUMBER = "matricule";
  
  /**
   * Identifiant JSON {@link #periodAsString}.
   */
  public static final String JSON_PERIOD_AS_STRING = "periodeChaine";
  
  /**
   * Identifiant JSON {@link #userName}.
   */
  public static final String JSON_USER_NAME = "nomUtilisateur";
  
  /**
   * Nom pour lecture humaine de la classe.
   */
  public static final String NAME = PayslipDto.NAME + " importable";

  /**
   * Nom au pluriel pour lecture humaine de la classe.
   */
  public static final String PLURAL_NAME = PayslipDto.PLURAL_NAME + "importables";
}
