package ci.gouv.dgbf.system.payroll.server.api;

import ci.gouv.dgbf.extension.server.service.api.AbstractIdentifiableFilter;
import ci.gouv.dgbf.extension.server.service.api.request.FilterDto;
import lombok.Getter;
import lombok.Setter;

/**
 * Cette classe représente le filtre de {@link FunctionMovementDto}.
 *
 * @author akm
 *
 */
@Getter
@Setter
public class FunctionMovementFilter extends AbstractIdentifiableFilter {
  /**
    * Identifiant du fonctionnaire.
    */
  String functionaryIdentifier;
  /**
    * Identifiant de la fonction.
    */
  String functionIdentifier;

  /**
    * Identifiant du suivi mouvement.
    */
  String movementTrackingIdentifier;

  /**
   * Cette méthode permet d'instantier un objet.
   *
   * @param dto filtre
   */
  public FunctionMovementFilter(FilterDto dto) {
    super(dto);
  }

  /**
   * Cette méthode permet d'instantier un objet.
   */
  public FunctionMovementFilter() {}

  @Override
  protected void doInitialize(FilterDto filter) {
    super.doInitialize(filter);
    functionaryIdentifier = getFunctionaryIdentifier(filter);
    functionIdentifier = getFunctionIdentifier(filter);
    movementTrackingIdentifier = getMovementTrackingIdentifier(filter);
  }

  @Override
  protected void toDto(FilterDto filter) {
    super.toDto(filter);
    setFunctionaryIdentifier(filter, functionaryIdentifier);
    setFunctionIdentifier(filter, functionIdentifier);
    setMovementTrackingIdentifier(filter, movementTrackingIdentifier);
  }

  /**
   * Cette méthode permet d'assigner l'identifiant du fonctionnaire.
   *
   * @param filter filtre
   * @param identifier identifiant
   */
  public static void setFunctionaryIdentifier(FilterDto filter, String identifier) {
    set(filter, JSON_FUNCTIONARY_IDENTIFIER, f -> f.getValueAsString(),
        f -> f.setValueAsString(identifier));
  }

  /**
   * Cette méthode permet d'obtenir l'identifiant du fonctionnaire.
   *
   * @param filter filtre
   * @return identifiant du fonctionnaire
   */
  public static String getFunctionaryIdentifier(FilterDto filter) {
    return get(filter, d -> d.getFieldValueAsStringByName(JSON_FUNCTIONARY_IDENTIFIER));
  }

  /**
   * Cette méthode permet d'assigner l'identifiant de la fonction.
   *
   * @param filter filtre
   * @param identifier identifiant
   */
  public static void setFunctionIdentifier(FilterDto filter, String identifier) {
    set(filter, JSON_FUNCTION_IDENTIFIER, f -> f.getValueAsString(),
        f -> f.setValueAsString(identifier));
  }

  /**
   * Cette méthode permet d'obtenir l'identifiant de la fonction.
   *
   * @param filter filtre
   * @return identifiant de la fonction
   */
  public static String getFunctionIdentifier(FilterDto filter) {
    return get(filter, d -> d.getFieldValueAsStringByName(JSON_FUNCTION_IDENTIFIER));
  }

  /**
   * Cette méthode permet d'assigner l'identifiant du suivi de mouvement.
   *
   * @param filter filtre
   * @param identifier identifiant
   */
  public static void setMovementTrackingIdentifier(FilterDto filter, String identifier) {
    set(filter, JSON_MOVEMENT_TRACKING_IDENTIFIER, f -> f.getValueAsString(),
        f -> f.setValueAsString(identifier));
  }

  /**
   * Cette méthode permet d'obtenir l'identifiant du suivi de mouvement.
   *
   * @param filter filtre
   * @return identifiant du suivi de mouvement
   */
  public static String getMovementTrackingIdentifier(FilterDto filter) {
    return get(filter, d -> d.getFieldValueAsStringByName(JSON_MOVEMENT_TRACKING_IDENTIFIER));
  }

  /**
   * Identifiant json champ {@link FunctionMovementFilter#functionaryIdentifier}.
   */
  public static final String JSON_FUNCTIONARY_IDENTIFIER = 
          FunctionMovementDto.JSON_FUNCTIONARY_IDENTIFIER;

  /**
   * Identifiant json champ {@link FunctionMovementFilter#functionIdentifier}.
   */
  public static final String JSON_FUNCTION_IDENTIFIER = 
          FunctionMovementDto.JSON_FUNCTION_IDENTIFIER;

  /**
   * Identifiant json champ {@link FunctionMovementFilter#movementTrackingIdentifier}.
   */
  public static final String JSON_MOVEMENT_TRACKING_IDENTIFIER = 
                FunctionMovementDto.JSON_MOVEMENT_TRACKING_IDENTIFIER;
  
}
