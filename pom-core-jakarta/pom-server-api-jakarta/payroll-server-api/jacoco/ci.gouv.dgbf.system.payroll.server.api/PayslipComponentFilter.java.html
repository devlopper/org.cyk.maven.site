package ci.gouv.dgbf.system.payroll.server.api;

import ci.gouv.dgbf.extension.server.service.api.AbstractIdentifiableFilter;
import ci.gouv.dgbf.extension.server.service.api.request.FilterDto;
import lombok.Getter;
import lombok.Setter;

/**
 * Cette classe représente le filtre de {@link PayslipComponentDto}.
 *
 * @author Christian
 *
 */
@Getter
@Setter
public class PayslipComponentFilter extends AbstractIdentifiableFilter {
  /**
   * Identifiant de {@link PayslipDto}.
   */
  String payslipIdentifier;

  /**
   * Identifiant de {@link PayrollComponentDto}.
   */
  String payrollComponentIdentifier;

  /**
   * Est un poste de gain.
   */
  Boolean isEarning;
  
  /**
   * Cette méthode permet d'instantier un objet.
   *
   * @param dto filtre
   */
  public PayslipComponentFilter(FilterDto dto) {
    super(dto);
  }

  /**
   * Cette méthode permet d'instantier un objet.
   */
  public PayslipComponentFilter() {}

  @Override
  protected void doInitialize(FilterDto filter) {
    super.doInitialize(filter);
    payslipIdentifier = getPayslipIdentifier(filter);
    payrollComponentIdentifier = getPayrollComponentIdentifier(filter);
    isEarning = getIsEarning(filter);
  }

  @Override
  protected void toDto(FilterDto filter) {
    super.toDto(filter);
    setPayslipIdentifier(filter, payslipIdentifier);
    setPayrollComponentIdentifier(filter, payrollComponentIdentifier);
    setIsEarning(filter, isEarning);
  }

  /**
   * Cette méthode permet d'assigner l'identifiant de {@link PayslipDto}.
   *
   * @param filter filtre
   * @param identifier identifiant
   */
  public static void setPayslipIdentifier(FilterDto filter, String identifier) {
    set(filter, JSON_PAYSLIP_IDENTIFIER, f -> f.getValueAsString(),
        f -> f.setValueAsString(identifier));
  }

  /**
   * Cette méthode permet d'obtenir l'identifiant de {@link PayslipDto}.
   *
   * @param filter filtre
   * @return identifiant de {@link PayslipDto}
   */
  public static String getPayslipIdentifier(FilterDto filter) {
    return get(filter, d -> d.getFieldValueAsStringByName(JSON_PAYSLIP_IDENTIFIER));
  }

  /**
   * Cette méthode permet d'assigner l'identifiant de {@link PayrollDto}.
   *
   * @param filter filtre
   * @param identifier identifiant de {@link PayrollDto}
   */
  public static void setPayrollComponentIdentifier(FilterDto filter, String identifier) {
    set(filter, JSON_PAYROLL_COMPONENT_IDENTIFIER, f -> f.getValueAsString(),
        f -> f.setValueAsString(identifier));
  }

  /**
   * Cette méthode permet d'obtenir l'identifiant de {@link PayrollDto}.
   *
   * @param filter filtre
   * @return identifiant de {@link PayrollDto}
   */
  public static String getPayrollComponentIdentifier(FilterDto filter) {
    return get(filter, d -> d.getFieldValueAsStringByName(JSON_PAYROLL_COMPONENT_IDENTIFIER));
  }
  
  /**
   * Cette méthode permet d'assigner est gain.
   *
   * @param filter {@link FilterDto}
   * @param isEarning est gain
   */
  public static void setIsEarning(FilterDto filter, Boolean isEarning) {
    set(filter, JSON_IS_EARNING, f -> f.getValueAsString(), f -> f.setValueAsBoolean(isEarning));
  }

  /**
   * Cette méthode permet d'obtenir est gain.
   *
   * @param filter {@link FilterDto}
   * @return vrai si est gain sinon faux
   */
  public static Boolean getIsEarning(FilterDto filter) {
    return get(filter, d -> d.getFieldValueAsBooleanByName(JSON_IS_EARNING));
  }

  /**
   * Identifiant json champ {@link PayslipComponentFilter#payslipIdentifier}.
   */
  public static final String JSON_PAYSLIP_IDENTIFIER = PayslipComponentDto.JSON_PAYSLIP_IDENTIFIER;

  /**
   * Identifiant json champ {@link PayslipComponentFilter#payrollComponentIdentifier}.
   */
  public static final String JSON_PAYROLL_COMPONENT_IDENTIFIER =
      PayslipComponentDto.JSON_PAYROLL_COMPONENT_IDENTIFIER;

  /**
   * Identifiant json {@link #isEarning}.
   */
  public static final String JSON_IS_EARNING = PayslipComponentDto.JSON_IS_EARNING;
}
