package ci.gouv.dgbf.system.payroll.server.api;

import ci.gouv.dgbf.extension.server.service.api.entity.AbstractIdentifiableAuditableDto;
import jakarta.json.bind.annotation.JsonbProperty;
import lombok.EqualsAndHashCode;
import lombok.Getter;
import lombok.Setter;

/**
 * Cette classe représente une paie.
 *
 * @author A.K.M
 *
 */
@Getter
@Setter
@EqualsAndHashCode(callSuper = true)
public class FunctionMovementDto extends AbstractIdentifiableAuditableDto {

  /**
   * Identifiant de la fonction.
   */
  @JsonbProperty(JSON_FUNCTION_IDENTIFIER)
  private String functionIdentifier;

  /**
   * Représentation en chaine de caractères  de la fonction.
   */
  @JsonbProperty(JSON_FUNCTION_AS_STRING)
  private String functionAsString;

  /**
   * Identifiant {@link MovementTrackingDto}.
   */
  @JsonbProperty(JSON_MOVEMENT_TRACKING_IDENTIFIER)
  private String movementTrackingIdentifier;

  /**
   * Représentation en chaine de caractères  du suivi de mouvement.
   */
  @JsonbProperty(JSON_MOVEMENT_TRACKING_AS_STRING)
  private String movementTrackingAsString;

  /**
   * Identifiant json de l'identifiant.
   */
  public static final String JSON_THIS_IDENTIFIER = "idMouvementFonction";

  /**
   * Identifiant json de la représentation en chaine de caractères.
   */
  public static final String JSON_THIS_AS_STRING = "mouvementFonctionChaine";

  /**
   * Identifiant json champ {@link #functionIdentifier}.
   */
  public static final String JSON_FUNCTION_IDENTIFIER = "idFonction";

  /**
   * Identifiant json champ {@link #functionAsString}.
   */
  public static final String JSON_FUNCTION_AS_STRING = "fonctionChaine";

  /**
   * Identifiant json champ {@link #movementTrackingIdentifier}.
   */
  public static final String JSON_MOVEMENT_TRACKING_IDENTIFIER = "idSuiviMouvement";

  /**
   * Identifiant json champ {@link #movementTrackingAsString}.
   */
  public static final String JSON_MOVEMENT_TRACKING_AS_STRING = "suiviMouvementChaine";

  /**
   * Nom pour lecture humaine de la classe.
   */
  public static final String NAME = "mouvement fonction";

  /**
   * Nom au pluriel pour lecture humaine de la classe.
   */
  public static final String PLURAL_NAME = "mouvements fonctions";
}
