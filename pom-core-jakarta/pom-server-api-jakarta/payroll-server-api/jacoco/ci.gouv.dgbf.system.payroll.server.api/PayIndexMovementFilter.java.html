package ci.gouv.dgbf.system.payroll.server.api;

import ci.gouv.dgbf.extension.server.service.api.AbstractIdentifiableFilter;
import ci.gouv.dgbf.extension.server.service.api.request.FilterDto;
import lombok.Getter;
import lombok.Setter;

/**
 * Cette classe représente le filtre de {@link PayIndexMovementDto}.
 *
 * @author akm
 *
 */
@Getter
@Setter
public class PayIndexMovementFilter extends AbstractIdentifiableFilter {
  /**
    * Identifiant de l'indice.
    */
  String indexIdentifier;

  /**
    * Identifiant du suivi mouvement.
    */
  String movementTrackingIdentifier;

  /**
   * Cette méthode permet d'instantier un objet.
   *
   * @param dto filtre
   */
  public PayIndexMovementFilter(FilterDto dto) {
    super(dto);
  }

  /**
   * Cette méthode permet d'instantier un objet.
   */
  public PayIndexMovementFilter() {}

  @Override
  protected void doInitialize(FilterDto filter) {
    super.doInitialize(filter);
    indexIdentifier = getIndexIdentifier(filter);
    movementTrackingIdentifier = getMovementTrackingIdentifier(filter);
  }

  @Override
  protected void toDto(FilterDto filter) {
    super.toDto(filter);
    setIndexIdentifier(filter, indexIdentifier);
    setMovementTrackingIdentifier(filter, movementTrackingIdentifier);
  }

  /**
   * Cette méthode permet d'assigner l'identifiant de l'indice.
   *
   * @param filter filtre
   * @param identifier identifiant
   */
  public static void setIndexIdentifier(FilterDto filter, String identifier) {
    set(filter, JSON_PAYINDEX_IDENTIFIER, f -> f.getValueAsString(),
        f -> f.setValueAsString(identifier));
  }

  /**
   * Cette méthode permet d'obtenir l'identifiant de l'indice.
   *
   * @param filter filtre
   * @return identifiant de l'indice
   */
  public static String getIndexIdentifier(FilterDto filter) {
    return get(filter, d -> d.getFieldValueAsStringByName(JSON_PAYINDEX_IDENTIFIER));
  }

  /**
   * Cette méthode permet d'assigner l'identifiant du suivi de mouvement.
   *
   * @param filter filtre
   * @param identifier identifiant
   */
  public static void setMovementTrackingIdentifier(FilterDto filter, String identifier) {
    set(filter, JSON_MOVEMENT_TRACKING_IDENTIFIER, f -> f.getValueAsString(),
        f -> f.setValueAsString(identifier));
  }

  /**
   * Cette méthode permet d'obtenir l'identifiant du suivi de mouvement.
   *
   * @param filter filtre
   * @return identifiant du suivi de mouvement
   */
  public static String getMovementTrackingIdentifier(FilterDto filter) {
    return get(filter, d -> d.getFieldValueAsStringByName(JSON_MOVEMENT_TRACKING_IDENTIFIER));
  }

  /**
   * Identifiant json champ {@link PayIndexMovementFilter#indexIdentifier}.
   */
  public static final String JSON_PAYINDEX_IDENTIFIER = PayIndexMovementDto
      .JSON_PAYINDEX_IDENTIFIER;

  /**
   * Identifiant json champ {@link PayIndexMovementFilter#movementTrackingIdentifier}.
   */
  public static final String JSON_MOVEMENT_TRACKING_IDENTIFIER = 
                PayIndexMovementDto.JSON_MOVEMENT_TRACKING_IDENTIFIER;
  
}
