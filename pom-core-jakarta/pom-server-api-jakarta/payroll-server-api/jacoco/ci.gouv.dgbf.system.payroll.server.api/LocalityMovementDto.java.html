package ci.gouv.dgbf.system.payroll.server.api;

import jakarta.json.bind.annotation.JsonbProperty;
import lombok.EqualsAndHashCode;
import lombok.Getter;
import lombok.Setter;

/**
 * Cette classe représente un mouvement de localité.
 *
 * @author stephane
 *
 */
@Getter
@Setter
@EqualsAndHashCode(callSuper = true)
@Deprecated(forRemoval = true)
public class LocalityMovementDto extends AbstractMovementTrackingIdentifiableAuditableDto {

  /**
   * Identifiant de la localité.
   */
  @JsonbProperty(JSON_LOCALITY_IDENTIFIER)
  private String localityIdentifier;

  /**
   * Représentation en chaine de caractères  de la localité.
   */
  @JsonbProperty(JSON_LOCALITY_AS_STRING)
  private String localityAsString;

  /**
   * Identifiant json de l'identifiant.
   */
  public static final String JSON_THIS_IDENTIFIER = "idMouvementLocalite";

  /**
   * Identifiant json de la représentation en chaine de caractères.
   */
  public static final String JSON_THIS_AS_STRING = "mouvementLocaliteChaine";

  /**
   * Identifiant json champ {@link #localityIdentifier}.
   */
  public static final String JSON_LOCALITY_IDENTIFIER = "idLocalite";

  /**
   * Identifiant json champ {@link #localityAsString}.
   */
  public static final String JSON_LOCALITY_AS_STRING = "localiteChaine";

  /**
   * Nom pour lecture humaine de la classe.
   */
  public static final String NAME = "mouvement localite";

  /**
   * Nom au pluriel pour lecture humaine de la classe.
   */
  public static final String PLURAL_NAME = "mouvements localites";
}
