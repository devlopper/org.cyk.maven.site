package ci.gouv.dgbf.system.payroll.server.api;

import jakarta.json.bind.annotation.JsonbProperty;
import lombok.EqualsAndHashCode;
import lombok.Getter;
import lombok.Setter;

/**
 * Cette classe représente un mouvement d'indice.
 *
 * @author A.K.M
 *
 */
@Getter
@Setter
@EqualsAndHashCode(callSuper = true)
public class PayIndexMovementDto extends AbstractMovementTrackingIdentifiableAuditableDto {

  /**
   * Identifiant de l'indice.
   */
  @JsonbProperty(JSON_PAYINDEX_IDENTIFIER)
  private String payIndexIdentifier;

  /**
   * Représentation en chaine de caractères de l'indice.
   */
  @JsonbProperty(JSON_PAYINDEX_AS_STRING)
  private String payIndexAsString;

  /**
   * Identifiant de l'emploi.
   */
  @JsonbProperty(JSON_JOB_IDENTIFIER)
  private String jobIdentifier;

  /**
   * Représentation en chaine de caractères de l'emploi.
   */
  @JsonbProperty(JSON_JOB_AS_STRING)
  private String jobAsString;

  /**
   * Identifiant du grade.
   */
  @JsonbProperty(JSON_GRADE_IDENTIFIER)
  private String gradeIdentifier;

  /**
   * Représentation en chaine de caractères du grade.
   */
  @JsonbProperty(JSON_GRADE_AS_STRING)
  private String gradeAsString;

  /**
   * Identifiant de l'échelon.
   */
  @JsonbProperty(JSON_ECHELON_IDENTIFIER)
  private String echelonIdentifier;

  /**
   * Représentation en chaine de caractères de l'échelon.
   */
  @JsonbProperty(JSON_ECHELON_AS_STRING)
  private String echelonAsString;

  /**
   * Identifiant de la classe.
   */
  @JsonbProperty(JSON_GRADE_CLASS_IDENTIFIER)
  private String gradeClassIdentifier;

  /**
   * Représentation en chaine de caractères de la classe.
   */
  @JsonbProperty(JSON_GRADE_CLASS_AS_STRING)
  private String gradeClassAsString;

  /**
   * Identifiant json de l'identifiant.
   */
  public static final String JSON_THIS_IDENTIFIER = "idMouvementIndice";

  /**
   * Identifiant json de la représentation en chaine de caractères.
   */
  public static final String JSON_THIS_AS_STRING = "mouvementIndiceChaine";

  /**
   * Identifiant json champ {@link #payIndexIdentifier}.
   */
  public static final String JSON_PAYINDEX_IDENTIFIER = "idIndice";

  /**
   * Identifiant json champ {@link #payIndexAsString}.
   */
  public static final String JSON_PAYINDEX_AS_STRING = "indiceChaine";

  /**
   * Identifiant json champ {@link #jobIdentifier}.
   */
  public static final String JSON_JOB_IDENTIFIER = "idEmploi";

  /**
   * Identifiant json champ {@link #jobAsString}.
   */
  public static final String JSON_JOB_AS_STRING = "emploiChaine";

  /**
   * Identifiant json champ {@link #gradeIdentifier}.
   */
  public static final String JSON_GRADE_IDENTIFIER = "idGrade";

  /**
   * Identifiant json champ {@link #gradeAsString}.
   */
  public static final String JSON_GRADE_AS_STRING = "gradeChaine";

  /**
   * Identifiant json champ {@link #echelonIdentifier}.
   */
  public static final String JSON_ECHELON_IDENTIFIER = "idEchelon";

  /**
   * Identifiant json champ {@link #echelonAsString}.
   */
  public static final String JSON_ECHELON_AS_STRING = "echelonChaine";

  /**
   * Identifiant json champ {@link #gradeClassIdentifier}.
   */
  public static final String JSON_GRADE_CLASS_IDENTIFIER = "idClasse";

  /**
   * Identifiant json champ {@link #gradeClassAsString}.
   */
  public static final String JSON_GRADE_CLASS_AS_STRING = "classeChaine";

  /**
   * Nom pour lecture humaine de la classe.
   */
  public static final String NAME = "mouvement indice";

  /**
   * Nom au pluriel pour lecture humaine de la classe.
   */
  public static final String PLURAL_NAME = "mouvements indices";
}
