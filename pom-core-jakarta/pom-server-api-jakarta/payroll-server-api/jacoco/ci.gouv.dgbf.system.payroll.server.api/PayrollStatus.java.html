package ci.gouv.dgbf.system.payroll.server.api;

import ci.gouv.dgbf.extension.core.Core;
import java.util.Collections;
import java.util.List;
import java.util.Set;
import java.util.stream.Stream;
import lombok.Getter;

/**
 * Cette classe représente les statuts de {@link PayrollDto}.
 *
 * @author A.K.M
 *
 */
@Getter
public enum PayrollStatus {

  /**
   * Paie créé.
   */
  CREATED(PayrollStatusCode.CREATED, "Créé", "Création"),
  /**
   * Paie Initialisé.
   */
  INITIALIZED(PayrollStatusCode.INITIALIZED, "Initialisé", "Initialisation"),
  /**
   * Paie controlé.
   */
  CONTROLED(PayrollStatusCode.CONTROLED, "Controlé", "Controle"),
  /**
   * Paie arbitragé.
   */
  ARBITRATED(PayrollStatusCode.ARBITRATED, "Arbitré", "Arbitrage"),
  /**
   * Paie Clôturé.
   */
  CLOSED(PayrollStatusCode.CLOSED, "Clôturé", "Clôture")

  ;

  private PayrollStatus(String code, String name, String action) {
    this.code = code;
    this.name = name;
    this.action = action;
  }

  private String code;
  private String name;
  private String action;
  private Set<PayrollStatus> previous;

  @Override
  public String toString() {
    return name;
  }

  /**
   * Cette méthode permet d'obtenir le statut par code.
   *
   * @param code code
   * @return statut
   */
  public static PayrollStatus getByCode(String code) {
    return Stream.of(PayrollStatus.values())
        .filter(status -> status.getCode().equals(code)).findFirst()
        .orElseThrow(IllegalArgumentException::new);
  }

  static {
    CREATED.previous = Collections.emptySet();
    INITIALIZED.previous = Core.getSet(List.of(CREATED, INITIALIZED));
    CONTROLED.previous = Core.getSet(List.of(INITIALIZED, CONTROLED));
    ARBITRATED.previous = Core.getSet(List.of(CONTROLED, ARBITRATED));
    CLOSED.previous = Core.getSet(List.of(ARBITRATED, CLOSED));
  }
}
