package ci.gouv.dgbf.system.payroll.server.api;

import ci.gouv.dgbf.extension.core.Core;
import java.util.Collections;
import java.util.List;
import java.util.Set;
import java.util.stream.Stream;
import lombok.Getter;

/**
 * Cette classe représente les statuts de {@link PayslipDto}.
 *
 * @author Christian
 *
 */
@Getter
public enum PayslipStatus {
  
  /**
   * Initialisée.
   */
  INITIALIZED(PayslipStatusCode.INITIALIZED, "Initialisée", "Initialisation"),
  
  /**
   * Transmis pour Controle.
   */
  TRANSMITTED_FOR_CONTROL(PayslipStatusCode.TRANSMITTED_FOR_CONTROL, 
        "Transmis pour contrôle", "Transmission pour contrôle"),
  
  /**
   * Controlée.
   */
  CONTROLLED(PayslipStatusCode.CONTROLLED, "Controlée", "Contrôle"),
  
  /**
   * Acceptée.
   */
  ACCEPTED(PayslipStatusCode.ACCEPTED, "Acceptée", "Acceptation"),
  
  /**
   * Validée.
   */
  VALIDATED(PayslipStatusCode.VALIDATED, "Validée", "Validation"),
  
  /**
   * Recalculée.
   */
  RECALCULATED(PayslipStatusCode.RECALCULATED, "Recalculée", "Recalcul"),
  
  /**
   * Arbitrée.
   */
  ARBITRATED(PayslipStatusCode.ARBITRATED, "Arbitrée", "Arbitrage"),
  
  /**
   * Clôturée.
   */
  CLOSED(PayslipStatusCode.CLOSED, "Clôturée", "Clôture")

  ;

  private PayslipStatus(String code, String name, String action) {
    this.code = code;
    this.name = name;
    this.action = action;
  }

  private String code;
  private String name;
  private String action;
  private Set<PayslipStatus> previous;

  @Override
  public String toString() {
    return name;
  }

  /**
   * Cette méthode permet d'obtenir le statut par code.
   *
   * @param code code
   * @return statut
   */
  public static PayslipStatus getByCode(String code) {
    return Stream.of(PayslipStatus.values())
        .filter(status -> status.getCode().equals(code)).findFirst()
        .orElseThrow(IllegalArgumentException::new);
  }

  static {
    INITIALIZED.previous = Collections.emptySet();
    TRANSMITTED_FOR_CONTROL.previous = Core.getSet(List.of(INITIALIZED));
    CONTROLLED.previous = Core.getSet(List.of(TRANSMITTED_FOR_CONTROL));
    ACCEPTED.previous = Core.getSet(List.of(CONTROLLED));
    RECALCULATED.previous = Core.getSet(List.of(ACCEPTED));
    VALIDATED.previous = Core.getSet(List.of(ACCEPTED, RECALCULATED));
    ARBITRATED.previous = Core.getSet(List.of(CONTROLLED));
    CLOSED.previous = Core.getSet(List.of(ARBITRATED));
  }
}
