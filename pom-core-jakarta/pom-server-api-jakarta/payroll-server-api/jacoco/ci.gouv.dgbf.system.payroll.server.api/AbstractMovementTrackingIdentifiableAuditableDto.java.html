package ci.gouv.dgbf.system.payroll.server.api;

  @JsonbProperty(JSON_DATE_REMINDER)

  @JsonbProperty(JSON_DATE_REMINDER_AS_STRING)
  private String dateReminderAsString;

   * {@link MovementStatus}.

  @JsonbProperty(JSON_STATUS)
  private MovementStatus status;

   * Représentation en chaine de caractères du statut de la situation de mouvement.
   */
  @JsonbProperty(JSON_STATUS_AS_STRING)

  public static final String JSON_MOVEMENT_TRACKING_IDENTIFIER = "idSuiviMouvement";

  /**

  public static final String JSON_MOVEMENT_TRACKING_AS_STRING = "suiviMouvementChaine";

  public static final String JSON_DATE_REMINDER_AS_STRING = "dateRappelChaine";
