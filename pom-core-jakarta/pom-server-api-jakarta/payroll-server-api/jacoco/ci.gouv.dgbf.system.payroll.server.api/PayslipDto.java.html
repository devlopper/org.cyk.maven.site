package ci.gouv.dgbf.system.payroll.server.api;

import ci.gouv.dgbf.extension.server.service.api.entity.AbstractIdentifiableAuditableDto;
import jakarta.json.bind.annotation.JsonbProperty;
import lombok.EqualsAndHashCode;
import lombok.Getter;
import lombok.Setter;

/**
 * Cette classe représente un bulletin de paie.
 *
 * @author Christian
 *
 */
@Getter
@Setter
@EqualsAndHashCode(callSuper = true)
public class PayslipDto extends AbstractIdentifiableAuditableDto implements PayslipContainerDto {

  /**
   * Identifiant {@link PayrollDto}.
   */
  @JsonbProperty(JSON_PAYROLL_IDENTIFIER)
  private String payrollIdentifier;

  /**
   * Représentation en chaine de caractère de {@link PayrollDto}.
   */
  @JsonbProperty(JSON_PAYROLL_AS_STRING)
  private String payrollAsString;

  /**
   * {@link PayslipStatus}.
   */
  @JsonbProperty(JSON_STATUS)
  private PayslipStatus status;

  /**
   * Représentation en chaine de caractères de {@link PayslipStatus}.
   */
  @JsonbProperty(JSON_STATUS_AS_STRING)
  private String statusAsString;

  /**
   * Identifiant {@link FunctionaryBudgetarySectionUnitDto}.
   */
  @JsonbProperty(JSON_FUNCTIONARY_BUDGETARY_SECTION_UNIT_IDENTIFIER)
  private String functionaryBudgetarySectionUnitIdentifier;

  /**
   * Représentation en chaine de caractère de {@link FunctionaryBudgetarySectionUnitDto}.
   */
  @JsonbProperty(JSON_FUNCTIONARY_BUDGETARY_SECTION_UNIT_AS_STRING)
  private String functionaryBudgetarySectionUnitAsString;

  /**
   * Représentation en chaine de caractère de {@link BudgetarySectionUnitDto}.
   */
  @JsonbProperty(JSON_BUDGETARY_SECTION_UNIT_AS_STRING)
  private String budgetarySectionUnitAsString;

  /**
   * Représentation en chaine de caractère de la section.
   */
  @JsonbProperty(JSON_BUDGETARY_SECTION_AS_STRING)
  private String budgetarySectionAsString;

  /**
   * Matricule.
   */
  @JsonbProperty(JSON_REGISTRATION_NUMBER)
  private String registrationNumber;

  /**
   * Nom et prénom(s).
   */
  @JsonbProperty(JSON_FIRST_NAME_AND_LAST_NAMES)
  private String firstNameAndLastNames;

  /**
   * Représentation en chaine de caractère de l'indice.
   */
  @JsonbProperty(JSON_PAY_INDEX_AS_STRING)
  private String payIndexAsString;

  /**
   * Représentation en chaine de caractère de la fonction.
   */
  @JsonbProperty(JSON_FUNCTION_AS_STRING)
  private String functionAsString;

  /**
   * Représentation en chaine de caractère du service.
   */
  @JsonbProperty(JSON_LOCALITY_AS_STRING)
  private String localityAsString;

  /**
   * Représentation en chaine de caractère du nombre d'enfants.
   */
  @JsonbProperty(JSON_CHILDREN_COUNT_AS_STRING)
  private String childrenCountAsString;

  /*-- Début définition des montants --*/

  /**
   * Représentation en chaine de caractères du montant brut.
   */
  @JsonbProperty(JSON_GROSS_AMOUNT_AS_STRING)
  private String grossAmountAsString;

  /**
   * Représentation en chaine de caractères du montant brut imposable.
   */
  @JsonbProperty(JSON_TAXABLE_GROSS_AMOUNT_AS_STRING)
  private String taxableGrossAmountAsString;

  /**
   * Représentation en chaine de caractères des éléments non soumis.
   */
  @JsonbProperty(JSON_NON_TAXABLE_AMOUNT_AS_STRING)
  private String nonTaxableAmountAsString;

  /**
   * Représentation en chaine de caractères du montant des retenues.
   */
  @JsonbProperty(JSON_DEDUCTIONS_AMOUNT_AS_STRING)
  private String deductionsAmountAsString;

  /**
   * Représentation en chaine de caractères du montant net.
   */
  @JsonbProperty(JSON_NET_AMOUNT_AS_STRING)
  private String netAmountAsString;

  /**
   * Représentation en chaine de caractères du montant initialisé.
   */
  @JsonbProperty(JSON_INITIALIZED_AMOUNT_AS_STRING)
  private String initializedAmountAsString;

  /**
   * Représentation en chaine de caractères du montant controlé.
   */
  @JsonbProperty(JSON_CONTROLED_AMOUNT_AS_STRING)
  private String controledAmountAsString;

  /**
   * Représentation en chaine de caractères du montant ajusté.
   */
  @JsonbProperty(JSON_AJUSTED_AMOUNT_AS_STRING)
  private String ajustedAmountAsString;

  /*-- Fin définition des montants --*/

  /**
   * Identifiant json de l'identifiant.
   */
  public static final String JSON_THIS_IDENTIFIER = "idBulletinPaie";

  /**
   * Identifiant json de la représentation en chaine de caractères.
   */
  public static final String JSON_THIS_AS_STRING = "bulletinPaieChaine";

  /**
   * Identifiant json {@link #payrollIdentifier}.
   */
  public static final String JSON_PAYROLL_IDENTIFIER = PayrollDto.JSON_THIS_IDENTIFIER;

  /**
   * Identifiant json {@link #payrollAsString}.
   */
  public static final String JSON_PAYROLL_AS_STRING = PayrollDto.JSON_THIS_AS_STRING;

  /**
   * Identifiant json champ {@link #status}.
   */
  public static final String JSON_STATUS = "statut";

  /**
   * Identifiant json champ {@link #statusAsString}.
   */
  public static final String JSON_STATUS_AS_STRING = "statutChaine";

  /**
   * Identifiant json {@link #functionaryBudgetarySectionUnitIdentifier}.
   */
  public static final String JSON_FUNCTIONARY_BUDGETARY_SECTION_UNIT_IDENTIFIER =
      "idFonctionnaireUniteSectionBudgetaire";

  /**
   * Identifiant json {@link #functionaryBudgetarySectionUnitAsString}.
   */
  public static final String JSON_FUNCTIONARY_BUDGETARY_SECTION_UNIT_AS_STRING =
      "fonctionnaireUniteSectionBudgetaireChaine";

  /**
   * Identifiant json {@link #budgetarySectionUnitAsString}.
   */
  public static final String JSON_BUDGETARY_SECTION_UNIT_AS_STRING = "uniteSectionBudgetaireChaine";

  /**
   * Identifiant json {@link #budgetarySectionAsString}.
   */
  public static final String JSON_BUDGETARY_SECTION_AS_STRING = "sectionBudgetaireChaine";

  /**
   * Identifiant json {@link #payIndexAsString}.
   */
  public static final String JSON_PAY_INDEX_AS_STRING = "indiceChaine";

  /**
   * Identifiant json {@link #functionAsString}.
   */
  public static final String JSON_FUNCTION_AS_STRING = "fonctionChaine";

  /**
   * Identifiant json {@link #localityAsString}.
   */
  public static final String JSON_LOCALITY_AS_STRING = "localiteChaine";

  /**
   * Identifiant json {@link #childrenCountAsString}.
   */
  public static final String JSON_CHILDREN_COUNT_AS_STRING = "nombreEnfantChaine";
  
  /**
   * Identifiant json {@link #registrationNumber}.
   */
  public static final String JSON_REGISTRATION_NUMBER = "matricule";

  /**
   * Identifiant json {@link #firstNameAndLastNames}.
   */
  public static final String JSON_FIRST_NAME_AND_LAST_NAMES = "nomPrenoms";

  /**
   * Nom pour lecture humaine de la classe.
   */
  public static final String NAME = "bulletin de paie";

  /**
   * Nom au pluriel pour lecture humaine de la classe.
   */
  public static final String PLURAL_NAME = "bulletins de paie";
}
