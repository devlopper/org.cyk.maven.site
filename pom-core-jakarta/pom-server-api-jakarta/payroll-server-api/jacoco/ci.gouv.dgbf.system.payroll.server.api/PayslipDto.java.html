package ci.gouv.dgbf.system.payroll.server.api;

import ci.gouv.dgbf.extension.server.service.api.entity.AbstractIdentifiableAuditableDto;
import jakarta.json.bind.annotation.JsonbProperty;
import lombok.EqualsAndHashCode;
import lombok.Getter;
import lombok.Setter;

/**
 * Cette classe représente un bulletin de salaire.
 *
 * @author Christian
 *
 */
@Getter
@Setter
@EqualsAndHashCode(callSuper = true)
public class PayslipDto extends AbstractIdentifiableAuditableDto {

  /**
   * Identifiant {@link PayrollDto}.
   */
  @JsonbProperty(JSON_PAYROLL_IDENTIFIER)
  private String payrollIdentifier;
  
  /**
   * Représentation en chaine de caractère de {@link PayrollDto}.
   */
  @JsonbProperty(JSON_PAYROLL_AS_STRING)
  private String payrollAsString;

  /**
   * Identifiant {@link EmployeeOrganizationDto}.
   */
  @JsonbProperty(JSON_EMPLOYEE_ORGANIZATION_IDENTIFIER)
  private String employeeOrganizationIdentifier;

  /**
   * Identifiant json de l'identifiant.
   */
  public static final String JSON_THIS_IDENTIFIER = "idBulletinSalaire";

  /**
   * Identifiant json de la représentation en chaine de caractères.
   */
  public static final String JSON_THIS_AS_STRING = "bulletinSalaireChaine";
  
  /**
   * Identifiant json {@link PayslipDto#payrollIdentifier}.
   */
  public static final String JSON_PAYROLL_IDENTIFIER = PayrollDto.JSON_THIS_IDENTIFIER;
  
  /**
   * Identifiant json {@link PayslipDto#payrollAsString}.
   */
  public static final String JSON_PAYROLL_AS_STRING = PayrollDto.JSON_THIS_AS_STRING;
  
  /**
   * Identifiant json {@link PayslipDto#employeeOrganizationIdentifier}.
   */
  public static final String JSON_EMPLOYEE_ORGANIZATION_IDENTIFIER = 
                      EmployeeOrganizationDto.JSON_THIS_IDENTIFIER;

  /**
   * Nom pour lecture humaine de la classe.
   */
  public static final String NAME = "bulletin de salaire";

  /**
   * Nom au pluriel pour lecture humaine de la classe.
   */
  public static final String PLURAL_NAME = "bulletins de salaire";
}
