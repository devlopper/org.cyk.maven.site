package ci.gouv.dgbf.system.payroll.server.api;

import ci.gouv.dgbf.extension.server.service.api.request.GetByIdentifierRequestDto;
import ci.gouv.dgbf.extension.server.service.api.request.GetManyRequestDto;

import jakarta.enterprise.context.ApplicationScoped;

import lombok.experimental.Accessors;

/**
 * Cette classe représente un client de {@link PayIndexMovementService}.
 *
 * @author akm
 *
 */
@ApplicationScoped
@Setter
@Accessors(chain = true, fluent = true)
public class PayIndexMovementClient extends AbstractClient<PayIndexMovementService>
    implements GetByIdentifier<PayIndexMovementDto>,
    GetMany<PayIndexMovementGetManyResponseDto>,
    DeleteByIdentifier<IdentifiableResponseDto> {

  @Override
  public PayIndexMovementClient service(PayIndexMovementService service) {
    return (PayIndexMovementClient) super.service(service);
  }

  /**
   * {@link PayIndexMovementService#create}.
   *
   * @param request requête
   * @return réponse
   */
  public CreateResponseDto create(PayIndexMovementCreateRequestDto request) {
    return new CreateExecutor(PayIndexMovementService.CREATE_IDENTIFIER)
        .execute(() -> service().create(request));
  }

  /**
   * {@link PayIndexMovementService#getMany}.
   *
   * @param request requête
   * @return réponse
   */
  public PayIndexMovementGetManyResponseDto getMany(GetManyRequestDto request) {
    return new GetOneExecutor<PayIndexMovementGetManyResponseDto>(
        PayIndexMovementGetManyResponseDto.class,
        PayIndexMovementService.GET_MANY_IDENTIFIER)
            .execute(() -> service().getMany(request));
  }

  /**
   * {@link PayIndexMovementService#getMany}.
   *
   * @param projection projection
   * @param filter filtre
   * @param page page
   * @param auditWho audit acteur
   * @param auditSession audit session
   * @return réponse
   */
  @Override
  public PayIndexMovementGetManyResponseDto getMany(ProjectionDto projection,
      FilterDto filter, PageDto page, String auditWho, String auditSession) {
    GetManyRequestDto request = new GetManyRequestDto();
    request.setProjection(projection);
    request.setFilter(filter);
    request.setPage(page);
    request.setAuditWho(auditWho);
    request.setAuditSession(auditSession);
    return getMany(request);
  }

  /**
   * {@link PayIndexMovementService#getOne}.
   *
   * @param request requête
   * @return réponse
   */
  public PayIndexMovementDto getOne(GetOneRequestDto request) {
    return new GetOneExecutor<PayIndexMovementDto>(PayIndexMovementDto.class,
        PayIndexMovementService.GET_ONE_IDENTIFIER)
            .execute(() -> service().getOne(request));
  }

  /**
   * {@link PayIndexMovementService#getOne}.
   *
   * @param projection projection
   * @param filter filtre
   * @param auditWho audit acteur
   * @param auditSession audit session
   * @return réponse
   */
  public PayIndexMovementDto getOne(ProjectionDto projection, FilterDto filter,
      String auditWho, String auditSession) {
    GetOneRequestDto request = new GetOneRequestDto();
    request.setProjection(projection);
    request.setFilter(filter);
    request.setAuditWho(auditWho);
    request.setAuditSession(auditSession);
    return getOne(request);
  }

  /**
   * {@link PayIndexMovementService#getByIdentifier}.
   *
   * @param request requête
   * @return groupe d'échéance
   */
  public PayIndexMovementDto getByIdentifier(GetByIdentifierRequestDto request) {
    return new GetOneExecutor<PayIndexMovementDto>(PayIndexMovementDto.class,
        PayIndexMovementService.GET_BY_IDENTIFIER_IDENTIFIER)
            .execute(() -> service().getByIdentifier(request));
  }

  /**
   * {@link PayIndexMovementService#getByIdentifier}.
   *
   * @param identifier identifiant
   * @param projection projection
   * @param auditWho audit acteur
   * @param auditSession audit session
   * @return réponse
   */
  @Override
  public PayIndexMovementDto getByIdentifier(String identifier, ProjectionDto projection,
      String auditWho, String auditSession) {
    GetByIdentifierRequestDto request = new GetByIdentifierRequestDto();
    request.setIdentifier(identifier);
    request.setProjection(projection);
    request.setAuditWho(auditWho);
    request.setAuditSession(auditSession);
    return getByIdentifier(request);
  }

  /**
   * {@link PayIndexMovementService#update}.
   *
   * @param request requête
   * @return réponse
   */
  public IdentifiableResponseDto update(PayIndexMovementUpdateRequestDto request) {
    return new IdentifiableExecutor(PayIndexMovementService.UPDATE_IDENTIFIER)
        .execute(() -> service().update(request));
  }

  /**
   * {@link PayIndexMovementService#delete}.
   *
   * @param request requête
   * @return réponse
   */
  public IdentifiableResponseDto delete(DeleteOneRequestDto request) {
    return new IdentifiableExecutor(PayIndexMovementService.DELETE_IDENTIFIER)
        .execute(() -> service().delete(request));
  }

  /**
   * {@link PayIndexMovementService#delete}.
   *
   * @param identifier identifiant
   * @param auditWho audit acteur
   * @param auditSession audit session
   * @return réponse
   */
  @Override
  public IdentifiableResponseDto deleteByIdentifier(String identifier, String auditWho,
      String auditSession) {
    DeleteOneRequestDto request = new DeleteOneRequestDto();
    request.setIdentifier(identifier);
    request.setAuditWho(auditWho);
    request.setAuditSession(auditSession);
    return delete(request);
  }
}
