package ci.gouv.dgbf.system.payroll.server.api;

import java.util.stream.Stream;
import lombok.Getter;

/**
 * Cette classe représente les types de mouvements de {@link MovementType }.
 *
 * @author akm
 *
 */
@Getter
public enum MovementType {

  /**
   * Fonction.
   */
  FUNCTION(MovementTypeCode.FUNCTION, "Fonction"),
  
  /**
   * Localité.
   */
  LOCALITY(MovementTypeCode.LOCALITY, "Localité"),
  
  /**
   * Service.
   */
  SERVICE(MovementTypeCode.SERVICE, "Service"),
  
  /**
   * Indice.
   */
  PAY_INDEX(MovementTypeCode.PAY_INDEX, "Indice")

  ;

  private MovementType(String code, String name) {
    this.code = code;
    this.name = name;
  }

  private String code;
  private String name;

  @Override
  public String toString() {
    return name;
  }

  /**
   * Cette méthode permet d'obtenir le type de movement par code.
   *
   * @param code code
   * @return type movement
   */
  public static MovementType getByCode(String code) {
    return Stream.of(MovementType.values())
        .filter(status -> status.getCode().equals(code)).findFirst()
        .orElseThrow(IllegalArgumentException::new);
  }
}
