package ci.gouv.dgbf.system.payroll.server.api;

import ci.gouv.dgbf.extension.server.service.api.AbstractIdentifiableFilter;
import ci.gouv.dgbf.extension.server.service.api.request.FilterDto;
import lombok.Getter;
import lombok.Setter;

/**
 * Cette classe représente le filtre de {@link EmployeeOrganizationDto}.
 *
 * @author Christian
 *
 */
@Getter
@Setter
public class EmployeeOrganizationFilter extends AbstractIdentifiableFilter {
  /**
    * Identifiant de {@link EmployeeDto}.
    */
  String employeeIdentifier;

  /**
    * Identifiant de {@link OrganizationDto}.
    */
  String organizationIdentifier;

  /**
   * Cette méthode permet d'instantier un objet.
   *
   * @param dto filtre
   */
  public EmployeeOrganizationFilter(FilterDto dto) {
    super(dto);
  }

  /**
   * Cette méthode permet d'instantier un objet.
   */
  public EmployeeOrganizationFilter() {}

  @Override
  protected void doInitialize(FilterDto filter) {
    super.doInitialize(filter);
    employeeIdentifier = getEmployeeIdentifier(filter);
    organizationIdentifier = getOrganizationIdentifier(filter);
  }

  @Override
  protected void toDto(FilterDto filter) {
    super.toDto(filter);
    setEmployeeIdentifier(filter, employeeIdentifier);
    setOrganizationIdentifier(filter, organizationIdentifier);
  }

  /**
   * Cette méthode permet d'assigner l'identifiant de {@link EmployeeDto}.
   *
   * @param filter filtre
   * @param identifier identifiant
   */
  public static void setEmployeeIdentifier(FilterDto filter, String identifier) {
    set(filter, JSON_EMPLOYEE_IDENTIFIER, f -> f.getValueAsString(),
        f -> f.setValueAsString(identifier));
  }

  /**
   * Cette méthode permet d'obtenir l'identifiant de {@link EmployeeDto}.
   *
   * @param filter filtre
   * @return identifiant de {@link EmployeeDto}
   */
  public static String getEmployeeIdentifier(FilterDto filter) {
    return get(filter, d -> d.getFieldValueAsStringByName(JSON_EMPLOYEE_IDENTIFIER));
  }

  /**
   * Cette méthode permet d'assigner l'identifiant de {@link EmployeeDto}.
   *
   * @param filter filtre
   * @param identifier identifiant
   */
  public static void setOrganizationIdentifier(FilterDto filter, String identifier) {
    set(filter, JSON_ORGANIZATION_IDENTIFIER, f -> f.getValueAsString(),
        f -> f.setValueAsString(identifier));
  }

  /**
   * Cette méthode permet d'obtenir l'identifiant de {@link EmployeeDto}.
   *
   * @param filter filtre
   * @return identifiant de {@link EmployeeDto}
   */
  public static String getOrganizationIdentifier(FilterDto filter) {
    return get(filter, d -> d.getFieldValueAsStringByName(JSON_ORGANIZATION_IDENTIFIER));
  }

  /**
   * Identifiant json champ {@link EmployeeOrganizationFilter#employeeIdentifier}.
   */
  public static final String JSON_EMPLOYEE_IDENTIFIER = 
                              EmployeeOrganizationDto.JSON_EMPLOYEE_IDENTIFIER;

  /**
   * Identifiant json champ {@link EmployeeOrganizationFilter#organizationIdentifier}.
   */
  public static final String JSON_ORGANIZATION_IDENTIFIER = 
                                EmployeeOrganizationDto.JSON_ORGANIZATION_IDENTIFIER;
  
}
