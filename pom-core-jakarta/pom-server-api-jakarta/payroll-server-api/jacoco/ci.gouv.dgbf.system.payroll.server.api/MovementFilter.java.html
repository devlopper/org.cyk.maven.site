package ci.gouv.dgbf.system.payroll.server.api;

import ci.gouv.dgbf.extension.server.service.api.AbstractIdentifiableFilter;
import ci.gouv.dgbf.extension.server.service.api.request.FilterDto;
import lombok.Getter;
import lombok.Setter;

/**
 * Cette classe représente le filtre de {@link FunctionMovementDto}.
 *
 * @author akm
 *
 */
@Getter
@Setter
public class MovementFilter extends AbstractIdentifiableFilter {
  
  /**
    * Identifiant du remunerable.
    */
  String remunerableIdentifier;
  
  /**
   * Identifiant du type de mouvement.
   */
  String typeIdentifier;
 
  /**
  * Identifiant de l'occurence d'élément de paie.
  */
  String payrollElementInstanceIdentifier;

  
  /**
   * Cette méthode permet d'instantier un objet.
   *
   * @param dto filtre
   */
  public MovementFilter(FilterDto dto) {
    super(dto);
  }

  /**
   * Cette méthode permet d'instantier un objet.
   */
  public MovementFilter() {}

  @Override
  protected void doInitialize(FilterDto filter) {
    super.doInitialize(filter);
    remunerableIdentifier = getRemunerableIdentifier(filter);
    typeIdentifier = getTypeIdentifier(filter);
    payrollElementInstanceIdentifier = getPayrollElementInstanceIdentifier(filter);
   
  }

  @Override
  protected void toDto(FilterDto filter) {
    super.toDto(filter);
    setRemunerableIdentifier(filter, remunerableIdentifier);
    setTypeIdentifier(filter, typeIdentifier);
    setPayrollElementInstanceIdentifier(filter, payrollElementInstanceIdentifier);
    
  }

  /**
   * Cette méthode permet d'assigner l'identifiant du remunerable.
   *
   * @param filter filtre
   * @param identifier identifiant
   */
  public static void setRemunerableIdentifier(FilterDto filter, String identifier) {
    set(filter, JSON_REMUNERABLE_IDENTIFIER, f -> f.getValueAsString(),
        f -> f.setValueAsString(identifier));
  }

  /**
   * Cette méthode permet d'obtenir l'identifiant du remunerable.
   *
   * @param filter filtre
   * @return identifiant du remunerable
   */
  public static String getRemunerableIdentifier(FilterDto filter) {
    return get(filter, d -> d.getFieldValueAsStringByName(JSON_REMUNERABLE_IDENTIFIER));
  }
  
  /**
   * Cette méthode permet d'assigner l'identifiant du type de mouvement.
   *
   * @param filter filtre
   * @param identifier identifiant
   */
  public static void setTypeIdentifier(FilterDto filter, String identifier) {
    set(filter, JSON_TYPE_IDENTIFIER, f -> f.getValueAsString(),
        f -> f.setValueAsString(identifier));
  }

  /**
   * Cette méthode permet d'obtenir l'identifiant du type de mouvement.
   *
   * @param filter filtre
   * @return identifiant du type de mouvement
   */
  public static String getTypeIdentifier(FilterDto filter) {
    return get(filter, d -> d.getFieldValueAsStringByName(JSON_TYPE_IDENTIFIER));
  }
  
  /**
   * Cette méthode permet d'assigner l'identifiant de l'occurence d'un élément de paie.
   *
   * @param filter filtre
   * @param identifier identifiant
   */
  public static void setPayrollElementInstanceIdentifier(FilterDto filter, String identifier) {
    set(filter, JSON_PAYROLL_ELEMENT_INSTANCE_IDENTIFIER, f -> f.getValueAsString(),
        f -> f.setValueAsString(identifier));
  }

  /**
   * Cette méthode permet d'obtenir l'identifiant de l'occurence d'un élément de paie.
   *
   * @param filter filtre
   * @return identifiant de l'occurence d'un élément de paie
   */
  public static String getPayrollElementInstanceIdentifier(FilterDto filter) {
    return get(filter, d -> d
        .getFieldValueAsStringByName(JSON_PAYROLL_ELEMENT_INSTANCE_IDENTIFIER));
  }

  /**
   * Identifiant json champ {@link MovementFilter#remunerableIdentifier}.
   */
  public static final String JSON_REMUNERABLE_IDENTIFIER = 
      RemunerableDto.JSON_IDENTIFIER;
  
  /**
   * Identifiant json champ {@link MovementFilter#typeIdentifier}.
   */
  public static final String JSON_TYPE_IDENTIFIER = 
      MovementTypeDto.JSON_IDENTIFIER;
  
  /**
   * Identifiant json champ {@link MovementFilter#payrollElementInstanceIdentifier}.
   */
  public static final String JSON_PAYROLL_ELEMENT_INSTANCE_IDENTIFIER = 
      PayrollElementInstanceDto.JSON_IDENTIFIER;

 
  
}
